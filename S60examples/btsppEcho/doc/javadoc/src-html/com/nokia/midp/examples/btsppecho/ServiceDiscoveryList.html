072        ServiceDiscoveryList(MIDletApplication midlet,
073                             String uuidString,
074                             int inquiryAccessCode)
075        {
076            super(TITLE, List.IMPLICIT);
077            this.midlet = midlet;
078            uuid = new UUID(uuidString, false);
079            this.inquiryAccessCode = inquiryAccessCode;
080    
081            openCommand = new Command("Open connection",
082                                      Command.SCREEN,
083                                      1);
084            searchCommand = new Command("Search", Command.SCREEN, 2);
085            logCommand = new Command("View log", Command.SCREEN, 3);
086            stopCommand = new Command("Stop", Command.SCREEN, 4);
087            backCommand = new Command("Back", Command.BACK, 5);
088            exitCommand = new Command("Exit", Command.EXIT, 0);
089    
090    
091            String property =
092                   LocalDevice.getProperty("bluetooth.sd.trans.max");
093            sdTransMax = Integer.parseInt(property);
094    
095            // create discovery agent
096            try
097            {
098                discoveryAgent =
099                    LocalDevice.getLocalDevice().getDiscoveryAgent();
100    
101                addCommand(logCommand);
102                addCommand(exitCommand);
103                addCommand(searchCommand);
104                setCommandListener(this);
105    
106                start();
107            }
108            catch(BluetoothStateException e)
109            {
110                midlet.serviceDiscoveryListFatalError(
111                          "Couldn't get a discovery agent: '" +
112                          e.getMessage() + "'");
113            }
114        }

154        public void init(int numConnectionsAlreadyOpen)
155        {
156            this.numConnectionsAlreadyOpen =
157                 numConnectionsAlreadyOpen;
158    
159            // stop any pending searches
160            if (inquiryInProgress ||
161                numServiceSearchesInProgress > 0)
162            {
163                cancelPendingSearches();
164            }
165    
166            // remove any old list elements
167            while (size() > 0)
168            {
169                delete(0);
170            }
171        }

177        }

180        public void commandAction(Command command, Displayable d)
181        {
182            if (command == logCommand)
183            {
184                midlet.serviceDiscoveryListViewLog(this);
185            }
186            else if (command == searchCommand &&
187                     !inquiryInProgress &&
188                     (numServiceSearchesInProgress == 0))
189            {
190                // new inquiry started
191    
192                removeCommand(openCommand);
193    
194                // remove old device lists
195                unsearchedRemoteDevices.removeAllElements();
196                for (Enumeration keys = matchingServiceRecords.keys();
197                     keys.hasMoreElements();)
198                {
199                    matchingServiceRecords.remove(keys.nextElement());
200                }
201    
202                // delete all old List items
203                while (size() > 0)
204                {
205                    delete(0);
206                }
207    
208                try
209                {
210                    // disable page scan and inquiry scan
211                    LocalDevice dev = LocalDevice.getLocalDevice();
212                    dev.setDiscoverable(DiscoveryAgent.NOT_DISCOVERABLE);
213    
214                    String iacString =
215                               LogScreen.inquiryAccessCodeString(inquiryAccessCode);
216                    LogScreen.log("startInquiry (" +
217                                  iacString + ")\n");
218    
219                    //startActivityIndicator();
220                    // this is non-blocking
221                    discoveryAgent.startInquiry(inquiryAccessCode, this);
222    
223                    setInquiryInProgress(true);
224                    addCommand(stopCommand);
225                    removeCommand(searchCommand);
226                }
227                catch (BluetoothStateException e)
228                {
229                    addCommand(searchCommand);
230                    removeCommand(stopCommand);
231                    midlet.serviceDiscoveryListError(
232                               "Error during startInquiry: '" +
233                               e.getMessage() + "'");
234                }
235            }
236            else if (command == stopCommand)
237            {
238                // stop searching
239                if (cancelPendingSearches())
240                {
241                    setInquiryInProgress(false);
242                    removeCommand(stopCommand);
243                    addCommand(searchCommand);
244                }
245            }
246            else if (command == exitCommand)
247            {
248                midlet.serviceDiscoveryListExitRequest();
249            }
250            else if (command == openCommand)
251            {
252                int size = this.size();
253                boolean[] flags = new boolean[size];
254                getSelectedFlags(flags);
255                Vector selectedServiceRecords = new Vector();
256                for (int i=0; i < size; i++)
257                {
258                    if (flags[i])
259                    {
260                        String key = getString(i);
261                        ServiceRecord rec =
262                            (ServiceRecord) matchingServiceRecords.get(key);
263                        selectedServiceRecords.addElement(rec);
264                    }
265                }
266    
267                // try to perform an open on selected items
268                if (selectedServiceRecords.size() > 0)
269                {
270                    String value = LocalDevice.getProperty(
271                                       "bluetooth.connected.devices.max");
272                    int maxNum = Integer.parseInt(value);
273    
274                    int total = numConnectionsAlreadyOpen +
275                                selectedServiceRecords.size();
276                    if (total > maxNum)
277                    {
278                        midlet.serviceDiscoveryListError(
279                                   "Too many selected. " +
280                                   "This device can connect to at most " +
281                                   maxNum + " other devices");
282                    }
283                    else
284                    {
285                        midlet.serviceDiscoveryListOpen(
286                                   selectedServiceRecords);
287                    }
288                }
289                else
290                {
291                    midlet.serviceDiscoveryListError(
292                              "Select at least one to open");
293                }
294            }
295            else if (command == backCommand)
296            {
297                midlet.serviceDiscoveryListBackRequest(backDisplayable);
298            }
299        }

301    
302        boolean cancelPendingSearches()
303        {
304            LogScreen.log("Cancel pending inquiries and searches\n");
305    
306            boolean everythingCancelled = true;
307    
308            if (inquiryInProgress)
309            {
310                // Note: The BT API (v1.0) isn't completely clear
311                // whether cancelInquiry is blocking or non-blocking.
312    
313                if (discoveryAgent.cancelInquiry(this))
314                {
315                    setInquiryInProgress(false);
316                }
317                else
318                {
319                    everythingCancelled = false;
320                }
321            }
322    
323            for (int i=0; i < transIds.size(); i++)
324            {
325                // Note: The BT API (v1.0) isn't completely clear
326                // whether cancelServiceSearch is blocking or
327                // non-blocking?
328    
329                Integer pendingId =
330                    (Integer) transIds.elementAt(i);
331                if (discoveryAgent.cancelServiceSearch(
332                                       pendingId.intValue()))
333                {
334                    transIds.removeElement(pendingId);
335                }
336                else
337                {
338                    everythingCancelled = false;
339                }
340            }
341            return everythingCancelled;
342        }

347        public void deviceDiscovered(RemoteDevice remoteDevice,
348                                     DeviceClass deviceClass)
349        {
350            LogScreen.log("deviceDiscovered: " +
351                          remoteDevice.getBluetoothAddress() +
352                          " major device class=" +
353                          deviceClass.getMajorDeviceClass() +
354                          " minor device class=" +
355                          deviceClass.getMinorDeviceClass() + "\n");
356    
357            // Note: The following check has the effect of only
358            // performing later service searches on phones.
359            // If you intend to run the MIDlet on some other device (e.g.
360            // handheld computer, PDA, etc. you have to add a check
361            // for them as well.) You might also refine the check
362            // using getMinorDeviceClass() to check only cellular phones.
363            boolean isPhone =
364                (deviceClass.getMajorDeviceClass() == 0x200);
365    
366            // Setting the following line to 'true' is a workaround
367            // for some early beta SDK device emulators. Set it
368            // to false when compiling the MIDlet for download to
369            // real MIDP phones!
370            boolean isEmulator = true; //false;
371    
372            if (isPhone || isEmulator)
373            {
374                unsearchedRemoteDevices.addElement(remoteDevice);
375            }
376        }

379        public void inquiryCompleted(int discoveryType)
380        {
381            LogScreen.log("inquiryCompleted: " +
382                          discoveryType + "\n");
383    
384            setInquiryInProgress(false);
385    
386            if (unsearchedRemoteDevices.size() == 0)
387            {
388                setTitle(TITLE);
389    
390                addCommand(searchCommand);
391                removeCommand(stopCommand);
392    
393                midlet.serviceDiscoveryListError(
394                           "No Bluetooth devices found");
395            }
396        }

399        public void servicesDiscovered(int transId,
400                                       ServiceRecord[] serviceRecords)
401        {
402            LogScreen.log("servicesDiscovered: transId=" +
403                          transId +
404                          " # serviceRecords=" +
405                          serviceRecords.length + "\n");
406    
407            // Remove+Add: ensure there is at most one open command
408            removeCommand(openCommand);
409            addCommand(openCommand);
410    
411            // Use the friendly name and/or bluetooth address
412            // to identify the matching Device + ServiceRecord
413            // (Note: Devices with different Bluetooth addresses
414            // might have the same friendly name, for example a
415            // device's default friendly name.)
416    
417            // there should only be one record
418            if (serviceRecords.length == 1)
419            {
420                RemoteDevice device = serviceRecords[0].getHostDevice();
421                String name = device.getBluetoothAddress();
422    
423                // This MIDlet only uses the first matching service
424                // record found for a particular device.
425                if (!matchingServiceRecords.containsKey(name))
426                {
427                    matchingServiceRecords.put(name, serviceRecords[0]);
428                    append(name, null);
429    
430                    // The List should have at least one entry,
431                    // before we add an open command.
432                    if (size() == 1)
433                    {
434                        addCommand(openCommand);
435                    }
436                }
437            }
438            else
439            {
440                midlet.serviceDiscoveryListError(
441                           "Error: Unexpected number (" +
442                           serviceRecords.length + ") of service records " +
443                           "in servicesDiscovered callback, transId=" +
444                           transId);
445            }
446        }

449        public void serviceSearchCompleted(int transId, int responseCode)
450        {
451            setTitle("New search");
452    
453            String responseCodeString =
454                LogScreen.responseCodeString(responseCode);
455            LogScreen.log("serviceSearchCompleted: transId=" +
456                        transId +
457                        " (" +
458                        responseCodeString + ")\n");
459    
460            // For any responseCode, decrement the counter
461            numServiceSearchesInProgress--;
462    
463            // remove the transaction id from the pending list
464            for (int i=0; i < transIds.size(); i++)
465            {
466                Integer pendingId = (Integer) transIds.elementAt(i);
467                if (pendingId.intValue() == transId)
468                {
469                    transIds.removeElement(pendingId);
470                    break;
471                }
472            }
473    
474            // all the searches have completed
475            if (!inquiryInProgress && (transIds.size() == 0))
476            {
477               removeCommand(stopCommand);
478               addCommand(searchCommand);
479    
480               if (matchingServiceRecords.size() == 0)
481               {
482                    midlet.serviceDiscoveryListError(
483                               "No matching services found");
484               }
485            }
486        }

489        // Interface Runnable
490        public void run()
491        {
492            Thread currentThread = Thread.currentThread();
493            int i = 0;
494    
495        running:
496            while (thread == currentThread)
497            {
498                synchronized (this)
499                {
500                    if (thread != currentThread)
501                    {
502                        break running;
503                    }
504                    else if (!inquiryInProgress)
505                    {
506                        doServiceSearch();
507                    }
508    
509                    if (inquiryInProgress ||
510                        numServiceSearchesInProgress > 0)
511                    {
512                        setTitle("Searching " + ACTIVITY[i]);
513                        if (++i >= ACTIVITY.length)
514                        {
515                            i = 0;
516                        }
517                    }
518    
519                    try
520                    {
521                        wait(WAIT_MILLIS);
522                    }
523                    catch (InterruptedException e)
524                    {
525                        // we can safely ignore this
526                    }
527                }
528            }
529        }

532        public void doServiceSearch()
533        {
534            if ((unsearchedRemoteDevices.size() > 0) &&
535                (numServiceSearchesInProgress < sdTransMax))
536            {
537                synchronized(this)
538                {
539                    RemoteDevice device =
540                        (RemoteDevice) unsearchedRemoteDevices
541                                           .elementAt(0);
542    
543                    UUID[] uuids = new UUID[1];
544                    uuids[0] = uuid;
545                    try
546                    {
547                        int[] attrSet = null; // default attrSet
548    
549                        numServiceSearchesInProgress++;
550    
551                        int transId = discoveryAgent.searchServices(
552                                                         attrSet,
553                                                         uuids,
554                                                         device,
555                                                         this);
556    
557                        LogScreen.log("starting service search on device=" +
558                                      device.getBluetoothAddress() +
559                                      " transId=" + transId + "\n");
560    
561                        transIds.addElement(new Integer(transId));
562                        unsearchedRemoteDevices.removeElementAt(0);
563                    }
564                    catch (BluetoothStateException e)
565                    {
566                        numServiceSearchesInProgress--;
567    
568                        midlet.serviceDiscoveryListError(
569                                   "Error, could not perform " +
570                                   "service search: '" +
571                                   e.getMessage());
572                    }
573                }
574            }
575        }

578        public void remove(Vector alreadyOpen)
579        {
580            if (size() > 0)
581            {
582                for (int i=0; i < alreadyOpen.size(); i++)
583                {
584                    // Bluetooth address of slave device that
585                    // we already have a connection open to
586                    String btAddress = (String) alreadyOpen.elementAt(i);
587    
588                    boolean found = false;
589                    for (int j = 0; j < size(); j++)
590                    {
591                        if (getString(j).equals(btAddress))
592                        {
593                            found = true;
594    
595                            // if the element we are about to
596                            // remove is selected, select something else
597                            if (getSelectedIndex() == j)
598                            {
599                                setSelectedIndex(j, false);
600                            }
601                            delete(j);
602                            break;
603                        }
604                    }
605                    if (found)
606                    {
607                        break;
608                    }
609                }
610            }
611        }
