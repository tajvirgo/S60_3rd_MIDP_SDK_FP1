002    package com.nokia.midp.examples.btsppecho.client;
003    
004    import java.io.IOException;
005    import javax.microedition.io.Connector;
006    import javax.microedition.io.StreamConnection;
007    import javax.microedition.io.StreamConnectionNotifier;
008    
009    import com.nokia.midp.examples.btsppecho.ClientForm;
010    import com.nokia.midp.examples.btsppecho.LogScreen;
011    
012    /**
013     * This class provides a connection service for bluetooth connections.
014     */
015    public class ConnectionService
016        implements Runnable
017    {
018        private final ClientForm listener;
019        private final String url;
020    
021        private StreamConnectionNotifier connectionNotifier = null;
022        private volatile boolean aborting;
023    
024    
025        public ConnectionService(String url,
026                                 ClientForm listener)
027        {
028            this.url = url;
029            this.listener = listener;
030    
031            LogScreen.log("ConnectionService: waiting to " +
032                          "accept connections on '" +
033                          url + "'\n");
034    
035            // start waiting for a connection
036            Thread thread = new Thread(this);
037            thread.start();
038        }
039    
040    
041        public String getClientURL()
042        {
043            return url;
044        }
045    
046    
047        public void close()
048        {
049            if (!aborting)
050            {
051                synchronized(this)
052                {
053                    aborting = true;
054                }
055    
056                // Ideally, one might want to give the run method's
057                // loop a chance to abort before calling the
058                // subsequent close, but the run loop is anyways
059                // likely to be sitting on the acceptAndOpen
060                // (i.e. blocked).
061    
062                try
063                {
064                    connectionNotifier.close();
065                }
066                catch (IOException e)
067                {
068                    // There is nothing very useful that
069                    // we can do for this case.
070                }
071            }
072        }
073    
074    
075        public void run()
076        {
077            aborting = false;
078    
079            try
080            {
081                connectionNotifier =
082                    (StreamConnectionNotifier) Connector.open(url);
083    
084                // It might useful in some cases to add a service to the
085                // 'Public Browse Group'. For example by doing something
086                // approximately as follows:
087                // -----------------------------------------------------
088                // Retrieve the service record template
089                // LocalDevice ld = LocalDevice.getLocalDevice();
090                // ServiceRecord rec = ld.getRecord(connectionNotifier);
091                // DataElement element =
092                //             new DataElement(DataElement.DATSEQ);
093                //
094                // The service class for PublicBrowseGroup (0x1002)
095                // is defined in the Bluetooth Assigned Numbers document.
096                // element.addElement(new DataElement(DataElement.UUID,
097                //                                    new UUID(0x1002)));
098                //
099                // Add to the public browse group:
100                // rec.setAttributeValue(0x0005, element);
101                // -----------------------------------------------------
102            }
103            catch (IOException e)
104            {
105                // ConnectionNotFoundException is an IOException
106                String errorMessage =
107                       "Error while starting ConnectionService: " +
108                       e.getMessage();
109    
110                listener.handleError(null, errorMessage);
111    
112                aborting = true;
113            }
114            catch (SecurityException e)
115            {
116                String errorMessage =
117                    "SecurityException while starting ConnectionService: " +
118                e.getMessage();
119    
120                listener.handleError(null, errorMessage);
121    
122                aborting = true;
123            }
124    
125            while (!aborting)
126            {
127                try
128                {
129                    // 1. wait to accept & open a new connection
130                    StreamConnection connection =
131                        (StreamConnection)
132                        connectionNotifier.acceptAndOpen();
133    
134                    LogScreen.log("ConnectionService: new connection\n");
135    
136                    // 2. create a handler to take care of
137                    // the new connection and inform
138                    // the listener
139                    if (!aborting)
140                    {
141                        ClientConnectionHandler handler =
142                            new ClientConnectionHandler(this,
143                                                        connection,
144                                                        listener);
145                        listener.handleAcceptAndOpen(handler);
146                    }
147    
148                    // One could consider exiting the
149                    // ConnectionService when the Client
150                    // reaches the maximum number of allowed
151                    // open connections. In that case (i.e.
152                    // when the maximum number of possible
153                    // connections is already open), the
154                    // ConnectionService will not be able
155                    // to accept any new connections and one
156                    // might possibly want to consider whether
157                    // or not the ConnectionService thread
158                    // could then be terminated.
159                    //
160                    // However, existing connections can also
161                    // be disconnected (e.g. the Server is
162                    // terminated or closes some/all of its
163                    // existing connections). In that case,
164                    // one may want to keep the
165                    // ConnectionService alive and running:
166                    // in order to accept later new connections
167                    // without the need to restart the
168                    // ConnectionService or MIDlet.
169                    //
170                    // (This MIDlet uses the latter approach.)
171                }
172                catch (IOException e)
173                {
174                    if (!aborting)
175                    {
176                        String errorMessage =
177                                   "IOException occurred during " +
178                                   "accept and open: " +
179                                   e.getMessage();
180    
181                        listener.handleError(null, errorMessage);
182                    }
183                }
184                catch (SecurityException e)
185                {
186                    if (!aborting)
187                    {
188                        String errorMessage =
189                                   "IOException occurred during " +
190                                   "accept and open: " +
191                                   e.getMessage();
192    
193                        listener.handleError(null, errorMessage);
194                    }
195                }
196            }
197        }
198    }
