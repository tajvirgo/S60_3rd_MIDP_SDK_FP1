002    package com.nokia.midp.examples.btsppecho;
003    
004    import javax.microedition.lcdui.*;
005    
006    
007    /**
008     * This class provides a form used for displaying text.
009     */
010    class TextScreen
011        extends Form
012        implements CommandListener
013    {
014        private final MIDletApplication midlet;
015        private final Displayable next;
016    
017    
018        TextScreen(MIDletApplication midlet,
019                   Displayable next,
020                   String title,
021                   String text,
022                   String closeLabel)
023        {
024            super(title);
025            this.midlet = midlet;
026            this.next = next;
027            append(text);
028            addCommand(new Command(closeLabel, Command.BACK, 1));
029            setCommandListener(this);
030        }
031    
032    
033        public void commandAction(Command c, Displayable d)
034        {
035            // The application code only adds a 'close' command.
036            midlet.textScreenClosed(next);
037        }
038    }
