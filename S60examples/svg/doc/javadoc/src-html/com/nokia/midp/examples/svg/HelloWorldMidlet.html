002    package com.nokia.midp.examples.svg;
003    
004    import javax.microedition.midlet.*;
005    import javax.microedition.lcdui.*;
006    
007    
008    /**
009     * Simple midlet that displays a static SVG file.
010     */
011    public class HelloWorldMidlet extends MIDlet implements CommandListener {
012    
013            /**
014             * Constructor.
015             */
016        public HelloWorldMidlet() {
017            
018            // *** create the SVG canvas
019            svgCanvas = new SvgCanvas1 (false);
020            svgCanvas.setCommandListener(this);
021            
022                    // *** grab a reference to the display
023            display = Display.getDisplay(this);
024            display.setCurrent(svgCanvas);
025            
026            // *** set up the midlet menu
027            int hotKey = 0;
028            
029            cmExit = new Command("Exit", Command.SCREEN, hotKey++);
030            svgCanvas.addCommand(cmExit);
031        }
032    
033    
034            /**
035             * Required midlet method.
036             */
037        public void startApp() {
038        }
039    
040    
041            /**
042             * Required midlet method.
043             */
044        public void pauseApp() {
045        }
046    
047        
048            /**
049             * Required midlet method.
050             */
051        public void destroyApp(boolean unconditional) {
052        }
053    
054    
055            /**
056             * Handle the various menu actions.
057             */
058        public void commandAction(Command c, Displayable s) {
059            if (c == cmExit) {
060                destroyApp(false);
061                notifyDestroyed();
062            }       
063        }
064        
065    
066            /*
067             * Private members
068             */
069    
070        private Display display;
071        private SvgCanvas1 svgCanvas;
072    
073        private Command cmExit;
074    
075    }
