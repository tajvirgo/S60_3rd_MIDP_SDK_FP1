003    
004    
005    import javax.microedition.midlet.*;
006    import javax.microedition.lcdui.*;
007    
008    
009    /**
010     * Simple midlet that displays a static SVG file.
011     */
012    public class LoadStaticMidlet extends MIDlet implements CommandListener {
013    
014            /**
015             * Constructor.
016             */
017        public LoadStaticMidlet() {
018            
019            // *** create the SVG canvas
020            svgCanvas = new SvgCanvas2 (false);
021            svgCanvas.setCommandListener(this);
022            
023                    // *** grab a reference to the display
024            display = Display.getDisplay(this);
025            display.setCurrent(svgCanvas);
026            
027            // *** set up the midlet menu
028            int hotKey = 0;
029            
030            cmZoomIn = new Command("Zoom In", Command.SCREEN, hotKey++);
031            svgCanvas.addCommand(cmZoomIn);
032            
033            cmZoomOut = new Command("Zoom Out", Command.SCREEN, hotKey++);
034            svgCanvas.addCommand(cmZoomOut);
035    
036            cmRotateIn = new Command("Rotate Counter", Command.SCREEN, hotKey++);
037            svgCanvas.addCommand(cmRotateIn);
038    
039            cmRotateOut = new Command("Rotate Clockwise", Command.SCREEN, hotKey++);
040            svgCanvas.addCommand(cmRotateOut);
041            
042            cmRestoreView = new Command("Restore View", Command.SCREEN, hotKey++);
043            svgCanvas.addCommand(cmRestoreView);
044            
045            cmExit = new Command("Exit", Command.SCREEN, hotKey++);
046            svgCanvas.addCommand(cmExit);
047        }
048    
049    
050            /**
051             * Required midlet method.
052             */
053        public void startApp() {
054        }
055    
056    
057            /**
058             * Required midlet method.
059             */
060        public void pauseApp() {
061        }
062    
063        
064            /**
065             * Required midlet method.
066             */
067        public void destroyApp(boolean unconditional) {
068        }
069    
070    
071            /**
072             * Handle the various menu actions.
073             */
074        public void commandAction(Command c, Displayable s) {
075            if (c == cmExit) {
076                destroyApp(false);
077                notifyDestroyed();
078            } else if ( c == cmZoomIn ) {
079                svgCanvas.zoomIn();
080            } else if ( c == cmZoomOut ) {
081                svgCanvas.zoomOut();
082            } else if ( c == cmRotateIn ) {
083                svgCanvas.rotateIn();
084            } else if ( c == cmRotateOut ) {
085                svgCanvas.rotateOut();
086            } else if ( c == cmRestoreView ) {
087                svgCanvas.restoreView();
088            }       
089        }
090        
091    
092            /*
093             * Private members
094             */
095    
096        private Display display;
097        private SvgCanvas2 svgCanvas;
098    
099        private Command cmZoomIn;
100        private Command cmZoomOut;
101        private Command cmRotateIn;
102        private Command cmRotateOut;
103        private Command cmRestoreView;
104        private Command cmExit;
105    
106    }
