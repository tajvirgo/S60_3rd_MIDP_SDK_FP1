002    package com.nokia.midp.examples.mms;
003    
004    import javax.microedition.lcdui.*;
005    
006    /**
007     * InfoScreen shows informational or error alert dialogs.
008     */
009    class InfoScreen
010    extends Alert
011    {
012            private final int timeInfo_duration = 3000;
013            private final int timeErr_duration = 5000;
014            private final String errTitle = "Error";
015            InfoScreen()
016            {
017                    super("InfoScreen");
018            }
019            /**
020             * Displays the information messages on the info screen.
021             * @param message the given message to be displayed
022             * @param display the display object
023             */
024            public void showInfo(String message, Display display)
025            {
026                    setTitle(null);
027                    setType(AlertType.INFO);
028                    setTimeout(timeInfo_duration);
029                    setString(message);
030                    display.setCurrent(this);
031            }
032    
033            /**
034             * Displays the error messages on the info screen.
035             * @param message the given message to be displayed
036             * @param display the display object
037             */
038            public void showError(String message, Display display)
039            {
040                    setTitle(errTitle);
041                    setType(AlertType.ERROR);
042                    setTimeout(timeErr_duration);
043                    setString(message);
044                    display.setCurrent(this);
045            }
046    
047    }
