003    
004    import javax.microedition.lcdui.*;
005    
006    
007    /**
008     * InstructionsScreen: Instructions Screen for game prototype.
009     */
010    class InstructionsScreen
011        extends Form
012        implements CommandListener
013    {
014        private final HawkMIDlet parent;
015        private final Command backCommand;
016    
017        private static final String instructions =
018            "This is an early prototype of an action game.\n\n" +
019            "In this version, all you can do is move around and fire.";
020    
021    
022        InstructionsScreen(HawkMIDlet parent)
023        {
024            super("Instructions");
025            this.parent = parent;
026    
027            append(new StringItem(null, instructions));
028    
029            backCommand = new Command("Back", Command.BACK, 1);
030            addCommand(backCommand);
031            setCommandListener(this);
032        }
033    
034    
035        public void commandAction(Command c, Displayable d)
036        {
037            // There is only one command: Back
038            parent.instructionsBack();
039        }
040    }
