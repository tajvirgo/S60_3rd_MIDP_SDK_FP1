002    package com.nokia.midp.examples.hawk;
003    
004    
005    import javax.microedition.midlet.*;
006    import javax.microedition.lcdui.*;
007    
008    
009    /**
010     * HawkMIDlet: Implements a simple game prototype that uses the MIDP 2.0 Game Canvas.
011     */
012    public class HawkMIDlet
013        extends MIDlet
014        implements Runnable
015    {
016        private MenuList menuList;
017        private InstructionsScreen instructionsScreen;
018        private HawkCanvas hawkCanvas;
019        private boolean initDone = false;
020    
021    
022        public HawkMIDlet()
023        {
024        }
025    
026    
027        public void startApp()
028        {
029            Displayable current = Display.getDisplay(this).getCurrent();
030            if (current == null)
031            {
032                // first time we've been called
033                Display.getDisplay(this).setCurrent(new SplashScreen(this));
034            }
035            else
036            {
037                if (current == hawkCanvas)
038                {
039                    hawkCanvas.start();
040                }
041                Display.getDisplay(this).setCurrent(current);
042            }
043        }
044    
045    
046        public void pauseApp()
047        {
048            Displayable current = Display.getDisplay(this).getCurrent();
049            if (current == hawkCanvas)
050            {
051                hawkCanvas.stop();
052            }
053        }
054    
055    
056        public void destroyApp(boolean unconditional)
057        {
058            if (hawkCanvas != null)
059            {
060                hawkCanvas.stop();   // kill its animation thread
061            }
062        }
063    
064    
065        private void quit()
066        {
067            destroyApp(false);
068            notifyDestroyed();
069        }
070    
071    
072        public void run()
073        {
074            init();
075        }
076    
077    
078        private synchronized void init()
079        {
080            if (!initDone)
081            {
082                menuList = new MenuList(this);
083                hawkCanvas = new HawkCanvas(this);
084                initDone = true;
085            }
086        }
087    
088    
089        void splashScreenPainted()
090        {
091            new Thread(this).start();  // start background initialization
092        }
093    
094    
095        void splashScreenDone()
096        {
097            init();   // if not already done
098            Display.getDisplay(this).setCurrent(menuList);
099        }
100    
101    
102        void menuListContinue()
103        {
104            Display.getDisplay(this).setCurrent(hawkCanvas);
105            hawkCanvas.start();
106        }
107    
108    
109        void menuListNewGame()
110        {
111            hawkCanvas.init();
112            Display.getDisplay(this).setCurrent(hawkCanvas);
113            hawkCanvas.start();
114        }
115    
116    
117        void menuListInstructions()
118        {
119            if (instructionsScreen == null)
120            {
121                instructionsScreen = new InstructionsScreen(this);
122            }
123            Display.getDisplay(this).setCurrent(instructionsScreen);
124        }
125    
126    
127        void menuListQuit()
128        {
129            quit();
130        }
131    
132    
133        void hawkCanvasMenu()
134        {
135            hawkCanvas.stop();
136            menuList.setGameActive(true);
137            Display.getDisplay(this).setCurrent(menuList);
138        }
139    
140    
141        void instructionsBack()
142        {
143            Display.getDisplay(this).setCurrent(menuList);
144        }
145    
146    
147        // method needed by lots of classes, shared by putting it here
148        static Image createImage(String filename)
149        {
150            Image image = null;
151            try
152            {
153                image = Image.createImage(filename);
154            }
155            catch (java.io.IOException e)
156            {
157                // just let return value be null
158            }
159            return image;
160        }
161    }
