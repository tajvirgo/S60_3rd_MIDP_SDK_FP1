026     * This class contains a list of discovered services.
027     */
028    class ServiceDiscoveryList
029        extends List
030        implements CommandListener,
031                   DiscoveryListener,
032                   Runnable
033    {
034        private final static String TITLE = "New search";
035        private final static int WAIT_MILLIS = 500; // milliseconds
036        private final static String[] ACTIVITY = { "",
037                                                   ".",
038                                                   "..",
039                                                   "...",
040                                                   "...." };

042        private final UUID uuid;
043        private final MIDletApplication midlet;
044        private final Command backCommand;
045        private final Command searchCommand;
046        private final Command openCommand;
047        private final Command stopCommand;
048        private final Command logCommand;
049        private final Command exitCommand;
050    
051        private final int sdTransMax;
052        private final int inquiryAccessCode;
053    
054        private DiscoveryAgent discoveryAgent;
055        private volatile boolean inquiryInProgress = false;
056        private volatile int numServiceSearchesInProgress = 0;
057        private volatile boolean aborting = false;
058        private volatile Thread thread;
059        private Displayable backDisplayable;
060    
061        private Vector unsearchedRemoteDevices = new Vector();
062        private Vector transIds = new Vector();
063        private Hashtable matchingServiceRecords = new Hashtable();
064    
065        private int numConnectionsAlreadyOpen = 0;

068        ServiceDiscoveryList(MIDletApplication midlet,
069                             String uuidString,
070                             int inquiryAccessCode)
071        {
072            super(TITLE, List.IMPLICIT);
073            this.midlet = midlet;
074            uuid = new UUID(uuidString, false);
075            this.inquiryAccessCode = inquiryAccessCode;
076    
077            openCommand = new Command("Open connection",
078                                      Command.SCREEN,
079                                      1);
080            searchCommand = new Command("Search", Command.SCREEN, 2);
081            logCommand = new Command("View log", Command.SCREEN, 3);
082            stopCommand = new Command("Stop", Command.SCREEN, 4);
083            backCommand = new Command("Back", Command.BACK, 5);
084            exitCommand = new Command("Exit", Command.EXIT, 0);
085    
086    
087            String property =
088                   LocalDevice.getProperty("bluetooth.sd.trans.max");
089            sdTransMax = Integer.parseInt(property);
090    
091            // create discovery agent
092            try
093            {
094                discoveryAgent =
095                    LocalDevice.getLocalDevice().getDiscoveryAgent();
096    
097                addCommand(logCommand);
098                addCommand(exitCommand);
099                addCommand(searchCommand);
100                setCommandListener(this);
101    
102                start();
103            }
104            catch(BluetoothStateException e)
105            {
106                midlet.serviceDiscoveryListFatalError(
107                           "Couldn't get a discovery agent: '" +
108                           e.getMessage() + "'");
109            }
110        }

115            this.backDisplayable = backDisplayable;
116            removeCommand(backCommand);

134        public void init(int numConnectionsAlreadyOpen)
135        {
136            this.numConnectionsAlreadyOpen =
137                 numConnectionsAlreadyOpen;
138    
139            // stop any pending searches
140            if (inquiryInProgress ||
141                numServiceSearchesInProgress > 0)
142            {
143                cancelPendingSearches();
144            }
145    
146            // remove any old list elements
147            while (size() > 0)
148            {
149                delete(0);
150            }
151        }
152    
153    
154        private synchronized void setInquiryInProgress(boolean bool)
155        {
156            inquiryInProgress = bool;
157        }

159    
160        public void commandAction(Command command, Displayable d)
161        {
162            if (command == logCommand)
163            {
164                midlet.serviceDiscoveryListViewLog(this);
165            }
166            else if (command == searchCommand &&
167                     !inquiryInProgress &&
168                     (numServiceSearchesInProgress == 0))
169            {
170                // new inquiry started
171    
172                removeCommand(openCommand);
173    
174                // remove old device lists
175                unsearchedRemoteDevices.removeAllElements();
176                for (Enumeration keys = matchingServiceRecords.keys();
177                     keys.hasMoreElements();)
178                {
179                    matchingServiceRecords.remove(keys.nextElement());
180                }
181    
182                // delete all old List items
183                while (size() > 0)
184                {
185                    delete(0);
186                }
187    
188                try
189                {
190                    // disable page scan and inquiry scan
191                    LocalDevice dev = LocalDevice.getLocalDevice();
192                    dev.setDiscoverable(DiscoveryAgent.NOT_DISCOVERABLE);
193    
194                    String iacString =
195                               LogScreen.inquiryAccessCodeString(inquiryAccessCode);
196                    LogScreen.log("startInquiry (" +
197                                  iacString + ")\n");
198    
199                    // this is non-blocking
200                    discoveryAgent.startInquiry(inquiryAccessCode, this);
201    
202                    setInquiryInProgress(true);
203                    addCommand(stopCommand);
204                    removeCommand(searchCommand);
205                }
206                catch (BluetoothStateException e)
207                {
208                    addCommand(searchCommand);
209                    removeCommand(stopCommand);
210                    midlet.serviceDiscoveryListError(
211                               "Error during startInquiry: '" +
212                               e.getMessage() + "'");
213                }
214            }
215            else if (command == stopCommand)
216            {
217                // stop searching
218                if (cancelPendingSearches())
219                {
220                    setInquiryInProgress(false);
221                    removeCommand(stopCommand);
222                    addCommand(searchCommand);
223                }
224            }
225            else if (command == exitCommand)
226            {
227                midlet.serviceDiscoveryListExitRequest();
228            }
229            else if (command == openCommand)
230            {
231                int size = this.size();
232                boolean[] flags = new boolean[size];
233                getSelectedFlags(flags);
234                Vector selectedServiceRecords = new Vector();
235                for (int i=0; i < size; i++)
236                {
237                    if (flags[i])
238                    {
239                        String key = getString(i);
240                        ServiceRecord rec =
241                            (ServiceRecord) matchingServiceRecords.get(key);
242                        selectedServiceRecords.addElement(rec);
243                    }
244                }
245    
246                // try to perform an open on selected items
247                if (selectedServiceRecords.size() > 0)
248                {
249                    String value = LocalDevice.getProperty(
250                                       "bluetooth.connected.devices.max");
251                    int maxNum = Integer.parseInt(value);
252    
253                    int total = numConnectionsAlreadyOpen +
254                                selectedServiceRecords.size();
255                    if (total > maxNum)
256                    {
257                        midlet.serviceDiscoveryListError(
258                                   "Too many selected. " +
259                                   "This device can connect to at most " +
260                                   maxNum + " other devices");
261                    }
262                    else
263                    {
264                        midlet.serviceDiscoveryListOpen(
265                                   selectedServiceRecords);
266                    }
267                }
268                else
269                {
270                    midlet.serviceDiscoveryListError(
271                              "Select at least one to open");
272                }
273            }
274            else if (command == backCommand)
275            {
276                midlet.serviceDiscoveryListBackRequest(backDisplayable);
277            }
278        }
279    
280    
281        boolean cancelPendingSearches()
282        {
283            LogScreen.log("Cancel pending inquiries and searches\n");
284    
285            boolean everythingCancelled = true;
286    
287            if (inquiryInProgress)
288            {
289                // Note: The BT API (v1.0) isn't completely clear
290                // whether cancelInquiry is blocking or non-blocking.
291    
292                if (discoveryAgent.cancelInquiry(this))
293                {
294                    setInquiryInProgress(false);
295                }
296                else
297                {
298                    everythingCancelled = false;
299                }
300            }
301    
302            for (int i=0; i < transIds.size(); i++)
303            {
304                // Note: The BT API (v1.0) isn't completely clear
305                // whether cancelServiceSearch is blocking or
306                // non-blocking?
307    
308                Integer pendingId =
309                    (Integer) transIds.elementAt(i);
310                if (discoveryAgent.cancelServiceSearch(
311                                       pendingId.intValue()))
312                {
313                    transIds.removeElement(pendingId);
314                }
315                else
316                {
317                    everythingCancelled = false;
318                }
319            }
320            return everythingCancelled;
321        }

326        public void deviceDiscovered(RemoteDevice remoteDevice,
327                                     DeviceClass deviceClass)
328        {
329            LogScreen.log("deviceDiscovered: " +
330                          remoteDevice.getBluetoothAddress() +
331                          " major device class=" +
332                          deviceClass.getMajorDeviceClass() +
333                          " minor device class=" +
334                          deviceClass.getMinorDeviceClass() + "\n");
335    
336            // Note: The following check has the effect of only
337            // performing later service searches on phones.
338            // If you intend to run the MIDlet on some other device (e.g.
339            // handheld computer, PDA, etc. you have to add a check
340            // for them as well.) You might also refine the check using
341            // the minor device class check only cellular phones.
342    
343            boolean isPhone =
344                (deviceClass.getMajorDeviceClass() == 0x200);
345    
346            // Setting the following line to 'true' is a workaround
347            // for some early beta SDK device emulators. Set it
348            // to false when compiling the MIDlet for download to
349            // real MIDP phones!
350            boolean isEmulator = false;
351    
352            if (isPhone || isEmulator)
353            {
354                unsearchedRemoteDevices.addElement(remoteDevice);
355            }
356        }

359        public void inquiryCompleted(int discoveryType)
360        {
361            LogScreen.log("inquiryCompleted: " +
362                          discoveryType + "\n");
363    
364            setInquiryInProgress(false);
365    
366            if (unsearchedRemoteDevices.size() == 0)
367            {
368                setTitle(TITLE);
369    
370                addCommand(searchCommand);
371                removeCommand(stopCommand);
372    
373                midlet.serviceDiscoveryListError(
374                           "No matching Bluetooth devices were found " +
375                           "that run the desired service");
376            }
377        }

380        public void servicesDiscovered(int transId,
381                                       ServiceRecord[] serviceRecords)
382        {
383            LogScreen.log("servicesDiscovered: transId=" +
384                          transId +
385                          " # serviceRecords=" +
386                          serviceRecords.length + "\n");
387    
388            // Remove+Add: ensure there is at most one open command
389            removeCommand(openCommand);
390            addCommand(openCommand);
391    
392            // Use the bluetooth address
393            // to identify the matching Device + ServiceRecord
394    
395            // there should only be one record
396            if (serviceRecords.length == 1)
397            {
398                RemoteDevice device = serviceRecords[0].getHostDevice();
399                String name = device.getBluetoothAddress();
400    
401                // This MIDlet only uses the first matching service
402                // record found for a particular device.
403                if (!matchingServiceRecords.containsKey(name))
404                {
405                    matchingServiceRecords.put(name, serviceRecords[0]);
406                    append(name, null);
407    
408                    // The List should have at least one entry,
409                    // before we add an open command.
410                    if (size() == 1)
411                    {
412                        addCommand(openCommand);
413                    }
414                }
415            }
416            else
417            {
418                midlet.serviceDiscoveryListError(
419                           "Error: Unexpected number (" +
420                           serviceRecords.length + ") of service records " +
421                           "in servicesDiscovered callback, transId=" +
422                           transId);
423            }
424        }

427        public void serviceSearchCompleted(int transId, int responseCode)
428        {
429            setTitle(TITLE);
430    
431            String responseCodeString =
432                LogScreen.responseCodeString(responseCode);
433            LogScreen.log("serviceSearchCompleted: transId=" +
434                        transId +
435                        " (" +
436                        responseCodeString + ")\n");
437    
438            // For any responseCode, decrement the counter
439            numServiceSearchesInProgress--;
440    
441            // remove the transaction id from the pending list
442            for (int i=0; i < transIds.size(); i++)
443            {
444                Integer pendingId = (Integer) transIds.elementAt(i);
445                if (pendingId.intValue() == transId)
446                {
447                    transIds.removeElement(pendingId);
448                    break;
449                }
450            }
451    
452            // all the searches have completed
453            if (!inquiryInProgress && (transIds.size() == 0))
454            {
455               removeCommand(stopCommand);
456               addCommand(searchCommand);
457    
458               if (matchingServiceRecords.size() == 0)
459               {
460                    midlet.serviceDiscoveryListError(
461                               "No matching services found");
462               }
463            }
464        }

467        // Interface Runnable
468        public void run()
469        {
470            Thread currentThread = Thread.currentThread();
471            int i = 0;
472    
473        running:
474            while (thread == currentThread)
475            {
476                synchronized (this)
477                {
478                    if (thread != currentThread)
479                    {
480                        break running;
481                    }
482                    else if (!inquiryInProgress)
483                    {
484                        doServiceSearch();
485                    }
486    
487                    if (inquiryInProgress ||
488                        numServiceSearchesInProgress > 0)
489                    {
490                        setTitle("Searching " + ACTIVITY[i]);
491                        if (++i >= ACTIVITY.length)
492                        {
493                            i = 0;
494                        }
495                    }
496    
497    
498                    try
499                    {
500                        wait(WAIT_MILLIS);
501                    }
502                    catch (InterruptedException e)
503                    {
504                        // we can safely ignore this
505                    }
506                }
507            }
508        }

510    
511        public void doServiceSearch()
512        {
513            if ((unsearchedRemoteDevices.size() > 0) &&
514                (numServiceSearchesInProgress < sdTransMax))
515            {
516                synchronized(this)
517                {
518                    RemoteDevice device =
519                        (RemoteDevice) unsearchedRemoteDevices
520                                           .elementAt(0);
521    
522                    UUID[] uuids = new UUID[1];
523                    uuids[0] = uuid;
524                    try
525                    {
526                        int[] attrSet = null; // default attrSet
527    
528                        numServiceSearchesInProgress++;
529    
530                        int transId = discoveryAgent.searchServices(
531                                                         attrSet,
532                                                         uuids,
533                                                         device,
534                                                         this);
535    
536                        LogScreen.log("starting service search on device=" +
537                                      device.getBluetoothAddress() +
538                                      " transId=" + transId + "\n");
539    
540                        transIds.addElement(new Integer(transId));
541                        unsearchedRemoteDevices.removeElementAt(0);
542                    }
543                    catch (BluetoothStateException e)
544                    {
545                        numServiceSearchesInProgress--;
546    
547                        midlet.serviceDiscoveryListError(
548                                   "Error, could not perform " +
549                                   "service search: '" +
550                                   e.getMessage());
551                    }
552                }
553            }
554        }

557        public void remove(Vector alreadyOpen)
558        {
559            if (size() > 0)
560            {
561                for (int i=0; i < alreadyOpen.size(); i++)
562                {
563                    // Bluetooth address of slave device that
564                    // we already have a connection open to
565                    String btAddress = (String) alreadyOpen.elementAt(i);
566    
567                    boolean found = false;
568                    for (int j = 0; j < size(); j++)
569                    {
570                        if (getString(j).equals(btAddress))
571                        {
572                            found = true;
573    
574                            // if the element we are about to
575                            // remove is selected, select something else
576                            if (getSelectedIndex() == j)
577                            {
578                                setSelectedIndex(j, false);
579                            }
580                            delete(j);
581                            break;
582                        }
583                    }
584                    if (found)
585                    {
586                        break;
587                    }
588                }
589            }
590        }
