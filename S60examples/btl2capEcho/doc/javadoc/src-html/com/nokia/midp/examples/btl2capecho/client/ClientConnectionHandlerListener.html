003    
004    /**
005     * Interface for a client bluetooth connection handler.
006     */
007    
008    public interface ClientConnectionHandlerListener
009    {
010        // The handler's accept and open of a new connection
011        // has occurred, but the I/O streams are not yet open.
012        // The I/O streams must be open to send or receive
013        // messages.
014        public void handleAcceptAndOpen(ClientConnectionHandler handler);
015    
016    
017        // The handler's I/O streams are now open and the
018        // handler can now be used to send and receive messages.
019        public void handleStreamsOpen(ClientConnectionHandler handler);
020    
021    
022        // Opening of the handler's I/O streams failed. The handler has
023        // closed any connections or streams that were open.
024        // The handler should not be used anymore,
025        // and should be discarded.
026        public void handleStreamsOpenError(ClientConnectionHandler handler,
027                                           String errorMessage);
028    
029    
030        // The handler got an inbound message.
031        public void handleReceivedMessage(
032                        ClientConnectionHandler handler,
033                        byte[] messageBytes);
034    
035    
036        // A message that had been previously queued for sending
037        // (identified by id) by the handler, has been sent successfully.
038        public void handleQueuedMessageWasSent(
039                        ClientConnectionHandler handler,
040                        Integer id);
041    
042    
043        // The handler has closed its connections and streams.
044        // The handler should not be used anymore, and should be discarded.
045        // Only handlers which have previously called 'handleOpen' may
046        // call 'handleClose', and only just once.
047        public void handleClose(ClientConnectionHandler handler);
048    
049    
050        // An error related to the handler occurred. The handler
051        // has closed the connection, and the handler should no
052        // longer be used.
053        public void handleErrorClose(ClientConnectionHandler handler,
054                                     String errorMessage);
055    
056    }
