004    /**
005     * An interface for a service connection handler listener.
006     */
007    public interface ServerConnectionHandlerListener
008    {
009        // The handler's open succeeded. It can now be used for sending
010        // and receiving messages.
011        public void handleOpen(ServerConnectionHandler handler);
012    
013    
014        // The hadler's open failed. It has closed any connections or
015        // streams that were open. The handler should not be used anymore,
016        // and should be discarded.
017        public void handleOpenError(ServerConnectionHandler handler,
018                                    String errorMessage);
019    
020    
021        // The handler got an inbound message.
022        public void handleReceivedMessage(
023                        ServerConnectionHandler handler,
024                        byte[] messageBytes);
025    
026    
027        // A message that had been previously queued for sending
028        // (identified by id) by the handler, has been sent successfully.
029        public void handleQueuedMessageWasSent(
030                        ServerConnectionHandler handler,
031                        Integer id);
032    
033    
034        // The handler has closed its connections and streams.
035        // The handler should not be used anymore, and should be discarded.
036        // Only handlers which have previously called 'handleOpen' may
037        // call 'handleClose', and only just once.
038        public void handleClose(ServerConnectionHandler handler);
039    
040    
041        // An error related to the handler occurred. The handler
042        // has closed the connection, and the handler should no
043        // longer be used.
044        public void handleErrorClose(ServerConnectionHandler handler,
045                                     String errorMessage);
046    
047    }
