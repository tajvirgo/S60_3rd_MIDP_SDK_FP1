002    package com.nokia.midp.examples.rms.shopping;
003    
004    import javax.microedition.lcdui.*;
005    
006    /**
007     * UI to edit the details of a ShoppingItem.
008     *
009     * The implementation is discussed in the package documentation for
010     * {@link com.nokia.midp.examples.rms.shopping} .
011     */
012    public class EditItemScreen extends Form {
013        /** Maximum number of characters in each text field. */
014        private static final int TEXT_FIELD_MAX_SIZE = 30;
015    
016        /** Reference to the parent midlet. */
017        private ShoppingListMIDlet midlet;
018    
019        /** Graphical component. */
020        private TextField itemName;
021        /** Graphical component. */
022        private TextField itemQuantity;
023    
024        /** The item being editted. */
025        private ShoppingItem item;
026    
027        /**
028         * Constructor.
029         * After construction, a call to editItem() is expected before the EditItemScreen is displayed.
030         *
031         *@param midlet The parent ShoppingListMIDlet.
032         */
033        public EditItemScreen(ShoppingListMIDlet midlet) {
034            super("RMS-Shopping List");
035            this.midlet = midlet;
036    
037            itemName = new TextField("Item", "", TEXT_FIELD_MAX_SIZE, TextField.ANY);
038            itemQuantity = new TextField("Quantity", "", TEXT_FIELD_MAX_SIZE, TextField.NUMERIC);
039    
040            append(itemName);
041            append(itemQuantity);
042            //the parent ShoppingListMIDlet acts as the CommandListener for
043            //this component, and also handles adding the Commands.
044        }
045    
046        /**
047         * Starts editing an item.  Any changes will be saved to the item itself,
048         * not a copy.
049         *
050         *@param item The item to edit;
051         */
052        public void editItem(ShoppingItem item) {
053            this.item = item;
054            itemName.setString(item.getName());
055            itemQuantity.setString(""+item.getQuantity());
056        }
057    
058        /**
059         * Saves the details into the original item, and returns it.
060         */
061        public ShoppingItem saveChanges() {
062            item.setDetails(itemName.getString(), Integer.parseInt(itemQuantity.getString()));
063            return item;
064        }
065    }
