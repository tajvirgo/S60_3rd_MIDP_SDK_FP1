002    package com.nokia.midp.examples.JSR75.PersonalController;
003    
004    import javax.microedition.midlet.MIDlet;
005    import javax.microedition.lcdui.*;
006    import javax.microedition.pim.*;
007    
008    /**
009    
010    GlobalServices gathers together services that are used by many classes in the similar way.
011    
012    
013    */
014    
015    public final class GlobalServices{
016    
017        public static final int SHOWN_FIELD_NOT_SUPPORTED = -1;
018        private static final String SHOWN_FIELD_NOT_SUPPORTED_TEXT = "-illegal field-";
019    
020        // Get the code of the field that is shown when PIMItems of
021        // the PIMList are listed in some list view.
022        public static int getRepresentativeField(int listTypeIndicator){
023            int representativeField = 0;
024            if(listTypeIndicator == PIM.CONTACT_LIST){
025                representativeField = Contact.FORMATTED_NAME;
026    
027            }else if(listTypeIndicator == PIM.EVENT_LIST){
028                representativeField = Event.SUMMARY;
029    
030            }else if(listTypeIndicator == PIM.TODO_LIST){
031                representativeField = ToDo.SUMMARY;
032            }
033            return representativeField;
034        }
035    
036    
037        // Creates new PIMItem but does not commit the item.
038        public static PIMItem createNewPIMItem(int listTypeIndicator, PIMList thePIMList){
039            PIMItem thePIMItem = null;
040            if(listTypeIndicator == PIM.CONTACT_LIST){
041                thePIMItem = ((ContactList) thePIMList).createContact();
042            }else if(listTypeIndicator == PIM.TODO_LIST){
043                thePIMItem = ((ToDoList) thePIMList).createToDo();
044            }else if(listTypeIndicator == PIM.EVENT_LIST){
045                thePIMItem = ((EventList) thePIMList).createEvent();
046            }
047            return thePIMItem;
048        }
049    
050    
051        // Deletes given PIMItem.
052        public static void deletePIMItem(int listTypeIndicator,
053                         PIMList currentPIMList,
054                         PIMItem thePIMItem) throws PIMException{
055            if(listTypeIndicator == PIM.CONTACT_LIST){
056                ((ContactList) currentPIMList).removeContact((Contact) thePIMItem);
057            }else if(listTypeIndicator == PIM.TODO_LIST){
058                ((ToDoList) currentPIMList).removeToDo((ToDo) thePIMItem);
059            }else if(listTypeIndicator == PIM.EVENT_LIST){
060                ((EventList) currentPIMList).removeEvent((Event) thePIMItem);
061            }
062        }
063    
064    
065        // Get the field value that is shown in the listings of PIMItems.
066        public static String getShownFieldData(int listTypeIndicator,
067                        PIMList currentPIMList,
068                        PIMItem thePIMItem){
069    
070            String shownFieldData = "";
071    
072            int shownFieldOfPIMItem = getRepresentativeField(listTypeIndicator);
073            if(!currentPIMList.isSupportedField(shownFieldOfPIMItem)){
074                shownFieldData = SHOWN_FIELD_NOT_SUPPORTED_TEXT;
075            }else if(thePIMItem.countValues(shownFieldOfPIMItem) > 0){
076                shownFieldData = thePIMItem.getString(shownFieldOfPIMItem, 0);
077                if(shownFieldData != null){
078                    shownFieldData = shownFieldData.trim();
079                }
080            }
081            return shownFieldData;
082        }
083    
084    
085        // Shows modal dialog.
086        // This is used for confirming delete operations etc.
087        public static void showModalDialog(MIDlet midlet,
088                           CommandListener cl,
089                           String title,
090                           String text,
091                           Command[] commands){
092            Alert alert = new Alert(title, text, null, AlertType.CONFIRMATION);
093            alert.setTimeout(Alert.FOREVER);
094            for(int i = 0; i < commands.length; i++){
095                alert.addCommand(commands[i]);
096            }
097            alert.setCommandListener(cl);
098            Display.getDisplay(midlet).setCurrent(alert);
099        }
100    
101    }
