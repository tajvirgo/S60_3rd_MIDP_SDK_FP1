051        public RepeatRulesScreen(PersonalController midlet, Displayable caller, int listTypeIndicator, PIMList currentPIMList, Event currentEvent) throws PIMException{
052            super(TITLE_TEXT);
053            this.midlet = midlet;
054            this.caller = caller;
055            this.currentPIMList=currentPIMList;
056            currentEventList=(EventList)currentPIMList;
057            this.listTypeIndicator = listTypeIndicator;
058            if(currentEvent !=null){
059                this.currentEvent = currentEvent;
060            } else{
061                currentEvent = currentEventList.createEvent();
062            }
063            this.usedPIMItem = currentEvent;
064    
065            addCommand(backCommand);
066            addCommand(newCommand);
067            addCommand(saveCommand);
068            addCommand(clearCommand);
069            setCommandListener(this);
070    
071            initializeTheForm();
072        }

074        // Handle the menu commands (CommandListener).
075        public void commandAction(Command command, Displayable displayable){
076            if (command == backCommand) {
077                    Display.getDisplay(midlet).setCurrent(caller);
078            }else if (command == clearCommand) {
079                try{
080                    this.usedPIMItem = GlobalServices.createNewPIMItem(listTypeIndicator, currentEventList);
081                    initializeTheForm();
082                } catch (Exception e) {
083                    ErrorReporter.reportError(e);
084                }
085    
086            }else if (command == newCommand) {
087                createNewRepeatRule();
088    
089            }else if (command == saveCommand) {
090                getFormFieldValues();
091                try {
092                    currentEvent.setRepeat(repeatRule);
093                    currentEvent.commit();
094                    initializeTheForm();
095                } catch (Exception e) {
096                    ErrorReporter.reportError(e);
097                }
098            }
099        }

101        // Get the values from the form fields (e.g. get the values the user has written).
102        private void getFormFieldValues(){
103            for(Enumeration e = formFields.keys(); e.hasMoreElements();){
104                Object formField = (Item) e.nextElement();
105                int field = ((Integer) formFields.get(formField)).intValue();
106    
107                String str;
108                int intValue;
109                int fieldValue;
110                switch(field){
111                    case RepeatRule.END:
112                        try {
113                            repeatRule.setDate(field, ((DateField)formField).getDate().getTime());
114                        } catch (Exception ex) {
115                            // The Date is null
116                        }
117                        break;
118                    // set int values
119                    case RepeatRule.FREQUENCY:
120                        str=((ChoiceGroup) formField).getString(((ChoiceGroup) formField).getSelectedIndex()) ;
121                        fieldValue = RepeatRuleData.getIntValues(str);
122                        if (fieldValue!=0) {
123                            repeatRule.setInt(field, fieldValue);
124                        }
125                        break;
126                    case RepeatRule.COUNT:
127                    case RepeatRule.INTERVAL:
128                    case RepeatRule.DAY_IN_MONTH:
129                    case RepeatRule.DAY_IN_YEAR:
130                        str=((TextField) formField).getString();
131                        try {
132                            intValue=Integer.parseInt(str);
133                            repeatRule.setInt(field,  intValue);
134                        } catch (NumberFormatException nfe) {
135                            // The String cannot be parsed as an int, e.g. it is NULL
136                        } catch (IllegalArgumentException iae) {
137                            // Field is not one of the the valid RepeatRule fields for this method,
138                            // or the value provided is not a valid value for the given field
139                        }
140                        break;
141                    case RepeatRule.MONTH_IN_YEAR:
142                        str=((ChoiceGroup) formField).getString(((ChoiceGroup) formField).getSelectedIndex()) ;
143                        fieldValue = RepeatRuleData.getIntValues(str);
144                        repeatRule.setInt(field, fieldValue);
145                        break;
146                    case RepeatRule.WEEK_IN_MONTH:
147                        str=((ChoiceGroup) formField).getString(((ChoiceGroup) formField).getSelectedIndex()) ;
148                        fieldValue = RepeatRuleData.getIntValues(str);
149                        repeatRule.setInt(field, fieldValue);
150                        break;
151                    case RepeatRule.DAY_IN_WEEK:
152                        str=((ChoiceGroup) formField).getString(((ChoiceGroup) formField).getSelectedIndex()) ;
153                        fieldValue = RepeatRuleData.getIntValues(str);
154                        repeatRule.setInt(field, fieldValue);
155                        break;
156    
157                }
158            }
159        }

162    // Create new Repeat Rule for currentEvent with predefined values
163    // This is not normal use, this only for demo purposes
164    // There are few examples of how to use this functionality.
165    // Some of those are in comments, to show different possibilities
166        private void createNewRepeatRule(){
167            if (currentEventList.isSupportedField(Event.SUMMARY)){
168                if( currentEvent.countValues(Event.SUMMARY) == 0 ){
169                    // There is no SUMMARY yet
170                    currentEvent.addString(Event.SUMMARY, PIMItem.ATTR_NONE, "New summary");
171                }
172                else if( currentEvent.countValues(Event.SUMMARY)!=0 ){
173                    // There is SUMMARY
174                }
175            } else {
176                // SUMMARY is not supported
177            }
178            // repeatRule.setInt(RepeatRule.FREQUENCY, RepeatRule.MONTHLY);
179            // repeatRule.setInt(RepeatRule.FREQUENCY, RepeatRule.WEEKLY);
180            repeatRule.setInt(RepeatRule.COUNT, 5);
181            // repeatRule.setInt(RepeatRule.INTERVAL, 7);
182            Calendar cal = Calendar.getInstance();
183            cal.set(Calendar.YEAR, 2005);
184            cal.set(Calendar.MONTH, Calendar.FEBRUARY);
185            cal.set(Calendar.DAY_OF_MONTH, 15);
186            cal.set(Calendar.HOUR_OF_DAY, 14);
187            cal.set(Calendar.MINUTE, 30);
188            Date date= cal.getTime();
189            repeatRule.setDate(RepeatRule.END, date.getTime());
190    
191            // repeatRule.setInt(RepeatRule.DAY_IN_WEEK, RepeatRule.FRIDAY);
192    
193            try {
194                initializeTheForm();
195            } catch (PIMException e) {
196            }
197        }

199        // Initializes the form fields.
200        // Gets the Event's field values and puts them to corresponding form
201        // fields so that user can view and change the values.
202        private void initializeTheForm() throws PIMException {
203            deleteAll();
204            formFields.clear();
205    
206            int[] repeatFields = repeatRule.getFields();
207            Item formField = null; // formField will be added to the form and shown.
208            formField = new ChoiceGroup("Frequency", ChoiceGroup.POPUP, RepeatRuleData.frequencyArray, null);
209            for (int i = 0; i < repeatFields.length; i++) {
210                if (repeatFields[i] ==RepeatRule.FREQUENCY){
211                    ((ChoiceGroup) formField).setSelectedIndex(
212                        RepeatRuleData.getArrayIndex(repeatRule.getInt(RepeatRule.FREQUENCY), RepeatRuleData.frequencyArray), true);
213                }
214            }
215            // Add the form field to the form.
216            if (formField != null) {
217                append(formField);
218                formFields.put(formField, new Integer(RepeatRule.FREQUENCY));
219            }
220    
221    /**
222            RepeatRule.DAILY supports   COUNT,INTERVAL, END
223            RepeatRule.WEEKLY supports  COUNT,INTERVAL, END, DAY_IN_WEEK
224            RepeatRule.MONTHLY supports COUNT,INTERVAL, END, DAY_IN_WEEK, WEEK_IN_MONTH, DAY_IN_MONTH
225            RepeatRule.YEARLY supports  COUNT,INTERVAL, END, MONTH_IN_YEAR, DAY_IN_WEEK, WEEK_IN_MONTH, DAY_IN_MONTH, DAY_IN_YEAR
226    */
227            // int[] supportedFields = currentEventList.getSupportedRepeatRuleFields(RepeatRule.DAILY);
228            // int[] supportedFields = currentEventList.getSupportedRepeatRuleFields(RepeatRule.WEEKLY);
229            // int[] supportedFields = currentEventList.getSupportedRepeatRuleFields(RepeatRule.MONTHLY);
230            int[] supportedFields = currentEventList.getSupportedRepeatRuleFields(RepeatRule.YEARLY);
231    
232            for (int i = 0; i < supportedFields.length; i++) {
233                formField = null; // formField will be added to the form and shown.
234                int supportedFieldDataType=supportedFields[i];
235                boolean repeatFound=false;
236                for (int j = 0; j < repeatFields.length; j++) {
237                    if (repeatFields[j]== supportedFields[i]){
238                        repeatFound=true;
239                        // break;
240                    }
241                }
242            int fieldValue;
243            String str=null;
244    
245                switch (supportedFieldDataType) {
246    
247                    case RepeatRule.COUNT:
248                        formField = new TextField("Count", null, INT_VALUE_SIZE, TextField.DECIMAL);
249                        if(repeatFound){
250                            fieldValue=repeatRule.getInt(supportedFieldDataType);
251                            ((TextField) formField).setString(Integer.toString(fieldValue));
252                        }
253                        break;
254    
255                    case RepeatRule.INTERVAL:
256                        formField = new TextField("Interval", null, INT_VALUE_SIZE, TextField.DECIMAL);
257                        if(repeatFound){
258                            fieldValue=repeatRule.getInt(supportedFieldDataType);
259                            ((TextField) formField).setString(Integer.toString(fieldValue));
260                        }
261                        break;
262    
263                    case RepeatRule.END:
264                        formField = new DateField("Ending date", DateField.DATE_TIME);
265                        if(repeatFound){
266                            ((DateField) formField).setDate(new Date(repeatRule.getDate(supportedFields[i])));
267                        }
268                        break;
269    
270                    case RepeatRule.MONTH_IN_YEAR:
271                        formField = new ChoiceGroup("Month in year", ChoiceGroup.POPUP, RepeatRuleData.monthInYearArray, null);
272                        if(repeatFound){
273                            ((ChoiceGroup) formField).setSelectedIndex(RepeatRuleData.getArrayIndex((repeatRule.getInt(supportedFields[i])), RepeatRuleData.monthInYearArray), true);
274                        }
275                        break;
276    
277                    case RepeatRule.WEEK_IN_MONTH:
278                        formField = new ChoiceGroup("Week in month", ChoiceGroup.POPUP, RepeatRuleData.weekInMonthArray, null);
279                        if(repeatFound){
280                            ((ChoiceGroup) formField).setSelectedIndex(RepeatRuleData.getArrayIndex((repeatRule.getInt(supportedFields[i])), RepeatRuleData.weekInMonthArray), true);
281                        }
282                        break;
283    
284                    case RepeatRule.DAY_IN_WEEK:
285                        formField = new ChoiceGroup("Day in week", ChoiceGroup.POPUP, RepeatRuleData.dayInWeekArray, null);
286                        if(repeatFound){
287                            ((ChoiceGroup) formField).setSelectedIndex(RepeatRuleData.getArrayIndex((repeatRule.getInt(supportedFields[i])), RepeatRuleData.dayInWeekArray), true);
288                        }
289                        break;
290    
291                    case RepeatRule.DAY_IN_MONTH:
292                        formField = new TextField("Day in month", null, INT_VALUE_SIZE, TextField.DECIMAL);
293                        if(repeatFound){
294                            fieldValue=repeatRule.getInt(supportedFieldDataType);
295                            ((TextField) formField).setString(Integer.toString(fieldValue));
296                        }
297                        break;
298    
299                    case RepeatRule.DAY_IN_YEAR:
300                        formField = new TextField("Day in year", null, INT_VALUE_SIZE, TextField.DECIMAL);
301                        if(repeatFound){
302                            fieldValue=repeatRule.getInt(supportedFieldDataType);
303                            ((TextField) formField).setString(Integer.toString(fieldValue));
304                        }
305                        break;
306    
307                }
308                if (formField != null) {
309                // Add the form field to the form.
310                    append(formField);
311                    // Put the field to the hashtable.
312                    formFields.put(formField, new Integer(supportedFieldDataType));
313                }
314            }
315        }
