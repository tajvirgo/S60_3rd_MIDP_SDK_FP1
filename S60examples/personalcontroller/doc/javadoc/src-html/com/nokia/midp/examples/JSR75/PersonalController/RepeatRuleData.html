002    package com.nokia.midp.examples.JSR75.PersonalController;
003    
004    import javax.microedition.pim.RepeatRule;
005    
006    /**
007     A container for RepeatRule data that provides an accessor method using a string to access an integer value.
008    */
009    public final class RepeatRuleData {
010    
011        public static final String[] frequencyArray     = {"no value", "Daily", "Weekly", "Monthly", "Yearly" };
012        public static final String[] monthInYearArray   ={ "no value", "January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December" };
013        public static final String[] weekInMonthArray   ={ "no value", "First", "Second", "Third", "Fourth", "Fifth", "Last", "Secondlast", "Thirdlast", "Fourthlast", "Fifthlast" };
014        public static final String[] dayInWeekArray     ={ "no value", "Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday" };
015    
016        public static int getArrayIndex(int intValue, String[] array){
017         for (int i=0; i<array.length ; i++){
018                if ( intValue==getIntValues(array[i]) ) return i;
019            }
020            return -1; // No int value for String founded
021        }
022    
023        public static int getIntValues(String str){
024            int intValue=-1; // default; String not founded
025            if ((str.compareTo("no value"))             == 0)   intValue = 0;
026        // RepeatRule.FREQUENCY
027            else if ((str.compareTo("Daily"))           == 0)   intValue = RepeatRule.DAILY;
028            else if ((str.compareTo("Weekly"))          == 0)   intValue = RepeatRule.WEEKLY;
029            else if ((str.compareTo("Monthly"))     == 0)   intValue = RepeatRule.MONTHLY;
030            else if ((str.compareTo("Yearly"))          == 0)   intValue = RepeatRule.YEARLY;
031        // RepeatRule.MONTH_IN_YEAR
032            else if ((str.compareTo("January"))     == 0)   intValue = RepeatRule.JANUARY;
033            else if ((str.compareTo("February"))        == 0)   intValue = RepeatRule.FEBRUARY;
034            else if ((str.compareTo("March"))           == 0)   intValue = RepeatRule.MARCH;
035            else if ((str.compareTo("April"))           == 0)   intValue = RepeatRule.APRIL;
036            else if ((str.compareTo("May"))             == 0)   intValue = RepeatRule.MAY;
037            else if ((str.compareTo("June"))            == 0)   intValue = RepeatRule.JUNE;
038            else if ((str.compareTo("July"))                == 0)   intValue = RepeatRule.JULY;
039            else if ((str.compareTo("August"))          == 0)   intValue = RepeatRule.AUGUST;
040            else if ((str.compareTo("September"))   == 0)   intValue = RepeatRule.SEPTEMBER;
041            else if ((str.compareTo("October"))     == 0)   intValue = RepeatRule.OCTOBER;
042            else if ((str.compareTo("November"))        == 0)   intValue = RepeatRule.NOVEMBER;
043            else if ((str.compareTo("December"))        == 0)   intValue = RepeatRule.DECEMBER;
044        // RepeatRule.WEEK_IN_MONTH
045            else if ((str.compareTo("First"))           == 0)   intValue = RepeatRule.FIRST;
046            else if ((str.compareTo("Second"))      == 0)   intValue = RepeatRule.SECOND;
047            else if ((str.compareTo("Third"))           == 0)   intValue = RepeatRule.THIRD;
048            else if ((str.compareTo("Fourth"))          == 0)   intValue = RepeatRule.FOURTH;
049            else if ((str.compareTo("Fifth"))           == 0)   intValue = RepeatRule.FIFTH;
050            else if ((str.compareTo("Last"))            == 0)   intValue = RepeatRule.LAST;
051            else if ((str.compareTo("Second last")) == 0)   intValue = RepeatRule.SECONDLAST;
052            else if ((str.compareTo("Third last"))      == 0)   intValue = RepeatRule.THIRDLAST;
053            else if ((str.compareTo("Fourth last")) == 0)   intValue = RepeatRule.FOURTHLAST;
054            else if ((str.compareTo("Fifth last"))      == 0)   intValue = RepeatRule.FIFTHLAST;
055        // RepeatRule.DAY_IN_WEEK
056            else if ((str.compareTo("Sunday"))      == 0)   intValue = RepeatRule.SUNDAY;
057            else if ((str.compareTo("Monday"))      == 0)   intValue = RepeatRule.MONDAY;
058            else if ((str.compareTo("Tuesday"))     == 0)   intValue = RepeatRule.TUESDAY;
059            else if ((str.compareTo("Wednesday"))   == 0)   intValue = RepeatRule.WEDNESDAY;
060            else if ((str.compareTo("Thursday"))        == 0)   intValue = RepeatRule.THURSDAY;
061            else if ((str.compareTo("Friday"))          == 0)   intValue = RepeatRule.FRIDAY;
062            else if ((str.compareTo("Saturday"))        == 0)   intValue = RepeatRule.SATURDAY;
063            return intValue;
064        }
065    }
