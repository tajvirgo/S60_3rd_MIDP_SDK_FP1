014    /**
015        ExportItem handles exporting items to files
016    */
017    
018    public class ExportItem extends Form implements CommandListener{
019    
020        private static final String TITLE_TEXT = "Export item";
021    
022        private final Command backCommand = new Command("Back", Command.BACK, 1);
023        private final Command selectCommand = new Command("Export", Command.OK, 1);
024    
025        private final int listTypeIndicator;
026        private PIMList currentPIMList;
027        private final Displayable caller;
028        private final PersonalController midlet;
029        private PIMItem usedPIMItem;
030    
031        private ChoiceGroup groupRoots;
032        private ChoiceGroup groupEncoding;
033        private TextField fileName;
034    
035        private final int DATA_SCREEN_TYPE_VIEW = 0;
036        private final int DATA_SCREEN_TYPE_SEARCH = 1;
037        private static final int STRING_VALUE_SIZE = 150;
038        private final int INDEX_OF_FILE_IN_URL = 8;
039    
040        public static final String[] encodingArray = { "UTF-8", "UTF-16", "ISO-8859-1", "US-ASCII" };
041        //ISO-8859-1, ISO-8859-6 (Arabic), UTF-8, UTF-16, UTF-16BE, UTF-16LE, US-ASCII, ISO-10646-UCS-2 (=UCS-2), Big5 (chinese traditional)
042        String schemePart = "file:///";
043    
044        public ExportItem(PersonalController midlet, Displayable caller, int listTypeIndicator, PIMList currentPIMList, PIMItem usedPIMItem) throws PIMException, IOException{
045    
046            super(TITLE_TEXT);
047            this.midlet = midlet;
048            this.caller = caller;
049            this.listTypeIndicator = listTypeIndicator;
050            this.currentPIMList=currentPIMList;
051            this.usedPIMItem=usedPIMItem;
052    
053            addCommand(backCommand);
054            addCommand(selectCommand);
055            setCommandListener(this);
056            InitializeTheForm();
057        }
058    
059        public void InitializeTheForm(){
060    
061            // Choose encodings
062            groupEncoding = new ChoiceGroup("Choose encoding", ChoiceGroup.POPUP, encodingArray, null);
063            append(groupEncoding);
064    
065            // This root query should be in thread to avoid potential deadlock
066            groupRoots = new ChoiceGroup( "Choose root", ChoiceGroup.POPUP );
067            Enumeration roots = FileSystemRegistry.listRoots();
068    
069            String privateDir = System.getProperty("fileconn.dir.private");
070    
071            // Append system property location as first choice if available.
072            if (privateDir != null)
073            {
074              groupRoots.append(privateDir.substring(INDEX_OF_FILE_IN_URL), null);
075            }
076            else
077            {
078                // Some implementations may not provide location to private directory
079                // in that case use an alternate writable location if available
080                // (e.g. graphics location)
081                String graphicsDir = System.getProperty("fileconn.dir.graphics");
082                if (graphicsDir != null)
083                {
084                  groupRoots.append(graphicsDir.substring(INDEX_OF_FILE_IN_URL), null);
085                }
086                // otherwise the user can slect any path from the root and file
087                // controls on the midlet...
088              }
089    
090    
091            for(; roots.hasMoreElements();){
092                groupRoots.append(roots.nextElement().toString(), null);
093            }
094            append(groupRoots);
095    
096            if(listTypeIndicator == PIM.CONTACT_LIST){
097             fileName = new TextField("File name", "vcard.txt", STRING_VALUE_SIZE, TextField.ANY);
098            } else {
099                fileName = new TextField("File name", "vcalendar.txt", STRING_VALUE_SIZE, TextField.ANY);
100            }
101            append(fileName);
102        }
103    
104        // Handle the menu commands (CommandListener).
105        public void commandAction(Command command, Displayable displayable){
106            if (command == backCommand) {
107                Display.getDisplay(midlet).setCurrent(caller);
108    
109            }else if (command == selectCommand) {
110                    String root=groupRoots.getString(groupRoots.getSelectedIndex());
111                    String file=fileName.getString();
112                    String url = schemePart + root + file;
113                    String encoding=groupEncoding.getString(groupEncoding.getSelectedIndex());
114                    ExportItemThread exportItemThread = new ExportItemThread(midlet, this, listTypeIndicator, currentPIMList, usedPIMItem, url, encoding);
115                    exportItemThread.start();
116            }
117        }
118    
119    }
120    
121    /**
122        A thread to avoid potential deadlock for operations that may block, such as networking.
123    */
124    class ExportItemThread extends Thread{
125        PersonalController midlet;
126        Displayable creator;
127        int listTypeIndicator;
128        PIMList currentPIMList;
129        PIMItem usedPIMItem;
130        String url;
131        String encoding;
132    
133        ExportItemThread(PersonalController midlet, Displayable creator, int listTypeIndicator, PIMList currentPIMList, PIMItem usedPIMItem, String url, String encoding){
134    
135            this.midlet = midlet;
136            this.creator = creator;
137            this.listTypeIndicator = listTypeIndicator;
138            this.currentPIMList=currentPIMList;
139            this.usedPIMItem=usedPIMItem;
140            this.url = url;
141            this.encoding=encoding;
142        }
143    
144        public void run(){
145            try{
146                FileConnection fconn = (FileConnection)Connector.open(url);
147                // If no exception is thrown, then the URI is valid, but the file may or may not exist.
148                if (!fconn.exists()) {
149                    fconn.create();  // create the file if it doesn't exist
150                }
151                OutputStream os = fconn.openOutputStream();
152                // Contacts are stored in vCard format, while calendar and to-do items are both stored in vCalendar format.
153                // Therefore data_formats[0] equals "vCard 2.1" with Contact List
154                // and "vCalendar 1.0" with ToDo and Event Lists.
155                String[] data_formats = PIM.getInstance().supportedSerialFormats(listTypeIndicator);
156                try {
157                    PIM.getInstance().toSerialFormat(usedPIMItem, os, encoding, data_formats[0]);
158                    os.close();
159                } catch (Exception e){
160                    ErrorReporter.reportError(e);
161                }
162                Display.getDisplay(midlet).setCurrent(creator);
163              } catch (SecurityException se){
164               // shows splash screen
165               String text = "Access Not Allowed";
166    
167               Alert splashScreen = new Alert(null, text, null, null);
168               Display.getDisplay(this.midlet).setCurrent(splashScreen);
169    
170             } catch (Exception e){
171               ErrorReporter.reportError(e);
172             }
173             // shows splash screen
174             String text = "Item Exported";
175    
176             Alert splashScreen = new Alert(null, text, null, null);
177             Display.getDisplay(this.midlet).setCurrent(splashScreen);
178        }
179    }
