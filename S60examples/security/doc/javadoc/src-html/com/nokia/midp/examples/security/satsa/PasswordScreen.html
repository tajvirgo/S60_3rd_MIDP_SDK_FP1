002    package com.nokia.midp.examples.security.satsa;
003    
004    import javax.microedition.lcdui.*;
005    
006    /**
007     * PasswordScreen allows the user to enter password to decrypt the message.
008     */
009    class PasswordScreen
010      extends Form
011      implements CommandListener
012    {
013      private final SATSAMIDlet midlet;
014      private final Command commandBack;
015      private final Command commandDecrypt;
016      private int index;
017      TextField passwordField;
018    
019      /**
020       * Creates a new PasswordScreen with "Back" and "Decrypt" commands.
021       * @param midlet the {@link SATSAMIDlet} object
022       */
023      PasswordScreen(SATSAMIDlet midlet)
024      {
025        super(null);
026        commandBack = new Command("Back", Command.BACK, 1);
027        commandDecrypt = new Command("Decrypt", Command.ITEM, 1);
028        this.midlet = midlet;
029        createForm();
030      }
031      /**
032       * Sets the index number for the password.
033       * @param index the index number
034       */
035      void setIndex(int index)
036      {
037        this.index = index;
038        createForm();
039      }
040      /**
041       * Creates the UI for the user to enter password to decrpt.
042       */
043      void createForm()
044      {
045        deleteAll();
046        setTitle("Enter password to decrypt.");
047        passwordField = new TextField("Password", "", 16, TextField.ANY);
048        append(passwordField);
049        addCommand(commandBack);
050        addCommand(commandDecrypt);
051        setCommandListener(this);
052      }
053    
054      /**
055       * Executes the command that the user invokes.
056       * @param c the command that the user sent
057       * @param d an object that has the capability of being placed
058       * on the display
059       */
060      public void commandAction(Command c, Displayable d)
061      {
062        if (c==commandBack)
063        {
064          midlet.showEncryptedMessage(index);
065        }
066        if (c==commandDecrypt)
067        {
068          midlet.showDecryptedMessage(index, passwordField.getString() );
069        }
070      }
071    }
