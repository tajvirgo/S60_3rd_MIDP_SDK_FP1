003    
004    import javax.microedition.lcdui.*;
005    
006    /**
007     * An InfoScreen is a screen that shows errors to the user.
008     */
009    class InfoScreen
010      extends Alert
011    {
012      /**
013       * Creates a new InfoScreen.
014       */
015      InfoScreen()
016      {
017        super("InfoScreen");
018      }
019    
020      /**
021       * Displays error messages.
022       * @param message The error message
023       * @param display The Displayable object
024       */
025      void showError(String message, Display display)
026      {
027        setTitle("Error");
028        setType(AlertType.ERROR);
029        setTimeout(5000);
030        setString(message);
031        display.setCurrent(this);
032      }
033    
034    }
