002    package com.nokia.midp.examples.security.satsa;
003    
004    import javax.microedition.lcdui.*;
005    
006    /**
007     * NewScreen creates UI for user to enter the message and the password need to
008     * be encrypted.
009     */
010    class NewScreen
011      extends Form
012      implements CommandListener
013    {
014    
015      private final SATSAMIDlet midlet;
016      private final Command commandBack;
017      private final Command commandSave;
018      private int index;
019      private TextField messageField;
020      private TextField passwordField;
021    
022      /**
023       * Creates a NewScreen with "Back" and "Save" commands.
024       * @param midlet the {@link SATSAMIDlet} object
025       */
026      NewScreen(SATSAMIDlet midlet)
027      {
028        super(null);
029        commandBack = new Command("Back", Command.BACK, 1);
030        commandSave = new Command("Save", Command.ITEM, 1);
031        this.midlet = midlet;
032        createForm();
033      }
034    
035      /**
036       * Sets the index number foe the message.
037       * @param index The index number to be set
038       */
039      void setIndex(int index)
040      {
041        this.index = index;
042        createForm();
043      }
044    
045      /**
046       * Creates the UI for the user to enter password and messages.
047       */
048      void createForm()
049      {
050        deleteAll();
051        setTitle("Add new message ");
052        messageField = new TextField("Message", "", 80, TextField.ANY);
053        passwordField = new TextField("Password", "", 16, TextField.ANY);
054        append(messageField);
055        append(passwordField);
056        addCommand(commandBack);
057        addCommand(commandSave);
058        setCommandListener(this);
059      }
060    
061      /**
062       * Executes the command that the user invokes.
063       * @param c the command that the user sent
064       * @param d an object that has the capability of being placed
065       * on the display
066       */
067      public void commandAction(Command c, Displayable d)
068      {
069        if (c==commandBack)
070        {
071          midlet.showMessageList();
072        }
073        if (c==commandSave)
074        {
075          midlet.addNewMessage(messageField.getString(), passwordField.getString() );
076        }
077      }
078    }
