002    package com.nokia.midp.examples.wma.chat;
003    
004    import javax.microedition.lcdui.*;
005    
006    /**
007     * Extends the WMAForm class and provides a screen to display messages that
008     * are sent and received.
009     *
010     * @version 1.0
011     */
012    public class ChatScreen extends WMAForm{
013       /** StringItem to display the text */
014       private StringItem chatText;
015       /** StringBuffer to append the newly received text to */
016       private StringBuffer strBuffer;
017    
018       /**
019        *   Sets up the screen with a StringItem and commands for getting
020        *   to the write and change peers screens.
021        *
022        *   @param screenHandler the current ScreenHandler
023        */
024       public ChatScreen (ScreenHandler screenHandler) {
025            super(screenHandler);
026            strBuffer = new StringBuffer();
027            chatText = new StringItem("", "");
028            chatText.setLayout(Item.LAYOUT_BOTTOM);
029            append(chatText);
030            addCommand(ScreenHandler.CMD_WRITE);
031            addCommand(ScreenHandler.CMD_PEERS);
032        }
033    
034        /**
035         *  Updates the ChatScreen with the sent or received message.
036         *
037         *  @param message the sent or received messsage
038         */
039        public void updateScreen(String message) {
040            strBuffer.insert(0, "\n"+message);
041            chatText.setText(strBuffer.toString());
042            chatText.setLayout(Item.LAYOUT_BOTTOM);
043        }
044    }
