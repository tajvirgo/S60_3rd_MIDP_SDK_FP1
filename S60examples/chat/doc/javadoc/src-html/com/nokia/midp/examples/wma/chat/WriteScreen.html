002    package com.nokia.midp.examples.wma.chat;
003    
004    import java.util.*;
005    import javax.microedition.lcdui.*;
006    
007    /**
008     * Extends the WMAForm class and provides a field in which to add text to be sent.
009     *
010     * @version 1.0
011     */
012    public class WriteScreen extends WMAForm{
013        /** String constant for TextField label */
014        private static final String TO_STRING = "To: ";
015        /** TextField for the text to be sent */
016        private TextField sendTextField;
017    
018       /**
019        *   Sets up the screen with a textfield in which to add text to send and adds
020        *   the Send and Cancel commands.
021        *
022        *   @param screenHandler the current ScreenHandler
023        */
024        public WriteScreen (ScreenHandler screenHandler) {
025            super(screenHandler);
026            sendTextField = new TextField("", "", ScreenHandler.TEXTFIELD_SIZE, TextField.ANY);
027            append(sendTextField);
028            addCommand(ScreenHandler.CMD_SEND);
029            addCommand(ScreenHandler.CMD_CANCEL);
030        }
031    
032        /**
033         *  Updates the textfield with newly selected peers.
034         *  Adds the peer names to the label of the TextField.
035         *
036         *  @param peerNumbers the currently selected peer numbers
037         */
038        public void updateScreen(Vector peerNumbers) {
039            String names = TO_STRING;
040            for ( Enumeration enumeration = peerNumbers.elements(); enumeration.hasMoreElements(); ) {
041                String number = (String)enumeration.nextElement();
042                if (names.equals(TO_STRING)) {
043                    names += number;
044                } else {
045                    names += "," + number;
046                }
047            }
048            sendTextField.setLabel(names);
049            sendTextField.setString("");
050        }
051    
052        /**
053         *  Returns the message to be sent.
054         *
055         *  @return a String containing the message to be sent
056         */
057        public String getMessageText() {
058            return sendTextField.getString();
059        }
060    }
