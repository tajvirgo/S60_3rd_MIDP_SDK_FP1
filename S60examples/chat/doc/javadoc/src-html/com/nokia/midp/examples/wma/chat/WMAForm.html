007     *  Extends the Form class to provide the common elements for all the screens.

014         *  Sets up the common elements of a Form for the all the screens.
015         *  Sets the Form title, adds an exit Command and sets the CommandListener.
016         *
017         *  @param screenHandler the current ScreenHandler
018         */
019        public WMAForm(ScreenHandler screenHandler) {

021            addCommand(ScreenHandler.CMD_EXIT);
022            setCommandListener(screenHandler);
