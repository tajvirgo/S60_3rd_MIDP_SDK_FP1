147      /**
148       * Sets the view
149       */
150      void setView()
151      {
152            // reset the locaiton and draw on top of the maze
153            locationSquare.setTranslation(locationx,
154                                                                      2 * WALL_HEIGHT + 3f, locationz);
155            // the square is rendered only in top view
156            locationSquare.setRenderingEnable(topView);
157            // the background is removed in top view
158            world.setBackground(topView ? null : background);
159            // sets the actual active camera
160            if (topView)
161            {
162              world.setActiveCamera(topCamera);
163            }
164            else
165            {
166              world.setActiveCamera(camera);
167            }
168      }

170      /**
171       * Builds the world
172       */
173      void init()
174      {
175            finished = false;
176            duration = 0;
177            setFullScreenMode(true);
178            // get a Graphics3D instance
179            g3d = Graphics3D.getInstance();
180            // build the world and the cameras
181            world = new World();
182            camera = new Camera();
183            topCamera = new Camera();
184            world.addChild(camera);
185            world.addChild(topCamera);
186            float w = (float)getWidth();
187            float h = (float)getHeight();
188    
189            // set the perspective
190            camera.setPerspective(60.f, w / h, 0.1f, 1000.f);
191            topCamera.setPerspective(60.f, w / h, 0.1f, 1000.f);
192    
193            // the top camer has a fixed transform looking from the top
194            Transform topCameraTransform = new Transform();
195            topCameraTransform.postRotate(90, -1f, 0f, 0f);
196            topCameraTransform.postTranslate(0f, 0f, 1.2f * MAZE_SIDE_LENGTH);
197            topCamera.setTransform(topCameraTransform);
198    
199            // Setup the background
200            background = new Background();
201            Image backgroundImage = MazeMIDlet.makeImage("/com/nokia/midp/examples/m3g/maze3d/content/background.png");
202            if (backgroundImage != null)
203            {
204              background.setImage(new Image2D(Image2D.RGB, backgroundImage));
205              background.setImageMode(Background.REPEAT, Background.REPEAT);
206            }
207            world.setBackground(background);
208    
209            createFloor();
210            createLocationSquare();
211            setUpMaze();
212            createStartEndMarks();
213            setView();
214    
215            // setup the initial location
216            locationx = maze.findStartLocationX();
217            locationz = maze.findStartLocationZ();
218            // look at the center
219            angley = -180f;
220    
221    
222            setupCamera();
223    
224            start();
225      }

227      /**
228       * Creates the label at the start and at the end of the maze
229       */
230      private void createStartEndMarks()
231      {
232            // The marks' appearance
233            Appearance startMarkAppearance = new Appearance();
234            Appearance endMarkAppearance = new Appearance();
235    
236            // The composite mode is ALPHA to show only the letters
237            // the background is hidden using the alpha layer
238            CompositingMode markCompositeMode = new CompositingMode();
239            markCompositeMode.setBlending(CompositingMode.ALPHA);
240            startMarkAppearance.setCompositingMode(markCompositeMode);
241            endMarkAppearance.setCompositingMode(markCompositeMode);
242    
243            // The label's text is built using a texture
244            Texture2D startMarkTexture = null;
245            Image startMarkTextureImage =
246                    MazeMIDlet.makeImage("/com/nokia/midp/examples/m3g/maze3d/content/start_label.png");
247            if (startMarkTextureImage != null)
248            {
249              startMarkTexture = new Texture2D(
250                      new Image2D(Image2D.RGBA, startMarkTextureImage));
251              // the texture is not repeated
252              startMarkTexture.setWrapping(Texture2D.WRAP_CLAMP,
253                                                                       Texture2D.WRAP_CLAMP);
254              startMarkTexture.setBlending(Texture2D.FUNC_REPLACE);
255              startMarkTexture.setFiltering(Texture2D.FILTER_NEAREST,
256                                                                            Texture2D.FILTER_NEAREST);
257              startMarkAppearance.setTexture(0, startMarkTexture);
258            }
259    
260            Texture2D endMarkTexture = null;
261            Image endMarkTextureImage =
262                    MazeMIDlet.makeImage("/com/nokia/midp/examples/m3g/maze3d/content/end_label.png");
263            if (endMarkTextureImage != null)
264            {
265              endMarkTexture = new Texture2D(
266                      new Image2D(Image2D.RGBA, endMarkTextureImage));
267              // the texture is not repeated
268              endMarkTexture.setWrapping(Texture2D.WRAP_CLAMP,
269                                                                     Texture2D.WRAP_CLAMP);
270              endMarkTexture.setBlending(Texture2D.FUNC_REPLACE);
271              endMarkTexture.setFiltering(Texture2D.FILTER_NEAREST,
272                                                                      Texture2D.FILTER_NEAREST);
273              endMarkAppearance.setTexture(0, endMarkTexture);
274            }
275            // create the start mesh
276            Plane startMarkPlane = maze.createStartMark();
277            Mesh startMarkMesh = startMarkPlane.createMesh();
278            startMarkMesh.setAppearance(0, startMarkAppearance);
279            // these are not pickable
280            startMarkMesh.setPickingEnable(false);
281            world.addChild(startMarkMesh);
282    
283            // creates the end mesh
284            Plane endMarkPlane = maze.createEndMark();
285            Mesh endMarkMesh = endMarkPlane.createMesh();
286            endMarkMesh.setAppearance(0, endMarkAppearance);
287            // is not pickable either
288            endMarkMesh.setPickingEnable(false);
289    
290            // Create a sequence of 4 keyframes
291            KeyframeSequence keyframes =
292                    new KeyframeSequence(4,3, KeyframeSequence.LINEAR);
293            keyframes.setDuration(12000);
294            keyframes.setRepeatMode(KeyframeSequence.LOOP);
295    
296            float[] trans = new float[4];
297            endMarkMesh.getOrientation(trans);
298            // create set the keyframes.
299            // Basically keep x and z and animate in the y axis
300            keyframes.setKeyframe(0, 0,
301                    new float[]{trans[0], trans[1], trans[2]});
302            keyframes.setKeyframe(1, 4000,
303                    new float[]{trans[0], trans[1] - WALL_HEIGHT / 4, trans[2]});
304            keyframes.setKeyframe(2, 8000,
305                    new float[]{trans[0], trans[1] + WALL_HEIGHT / 4, trans[2]});
306            keyframes.setKeyframe(3, 12000,
307                    new float[]{trans[0], trans[1], trans[2]});
308            AnimationTrack animationTrack =
309                    new AnimationTrack(keyframes, AnimationTrack.TRANSLATION);
310    
311            AnimationController anim = new AnimationController();
312            animationTrack.setController(anim);
313    
314            // add the animation track
315            endMarkMesh.addAnimationTrack(animationTrack);
316    
317            world.addChild(endMarkMesh);
318      }

320      /**
321       * Creates the floor
322       */
323      private void createFloor()
324      {
325            float floorSide = MAZE_SIDE_LENGTH / 2;
326    
327            // define the location and size of the floor
328            Transform floorTransform = new Transform();
329            floorTransform.postRotate(90.0f, -1.0f, 0.0f, 0.0f);
330            floorTransform.postScale(floorSide, floorSide, 1.0f);
331    
332            // The floor appearance. Basically a texture repeated many times
333            Appearance floorAppearance = new Appearance();
334            // The floor needs that perspective correction is enabled
335            PolygonMode floorPolygonMode = new PolygonMode();
336            floorPolygonMode.setPerspectiveCorrectionEnable(true);
337            floorAppearance.setPolygonMode(floorPolygonMode);
338    
339            // load the texture
340            Texture2D floorTexture = null;
341            Image floorTextureImage = MazeMIDlet.makeImage("/com/nokia/midp/examples/m3g/maze3d/content/floor.png");
342            if (floorTextureImage != null)
343            {
344              floorTexture = new Texture2D(
345                      new Image2D(Image2D.RGB, floorTextureImage));
346              // the texture is repeated many times
347              floorTexture.setWrapping(Texture2D.WRAP_REPEAT,
348                                                               Texture2D.WRAP_REPEAT);
349              floorTexture.setBlending(Texture2D.FUNC_REPLACE);
350              floorTexture.setFiltering(Texture2D.FILTER_LINEAR,
351                                                                    Texture2D.FILTER_NEAREST);
352              floorAppearance.setTexture(0, floorTexture);
353            }
354            Plane floor = new Plane(floorTransform, 10);
355    
356            // build the mesh
357            Mesh floorMesh = floor.createMesh();
358            floorMesh.setAppearance(0, floorAppearance);
359            // the floor is not pickable
360            floorMesh.setPickingEnable(false);
361            // add to the world
362            world.addChild(floorMesh);
363      }

402      /**
403       * Sets up the maze
404       */
405      private void setUpMaze()
406      {
407            // the walls need perspection correction enabled
408            PolygonMode wallPolygonMode = new PolygonMode();
409            wallPolygonMode.setPerspectiveCorrectionEnable(true);
410    
411            // build the wall semi transparent appearance
412            wallClearAppearance.setPolygonMode(wallPolygonMode);
413            // this is to make a wall semi transparent
414            CompositingMode wallClearCompositeMode = new CompositingMode();
415            wallClearCompositeMode.setBlending(CompositingMode.ALPHA_ADD);
416            wallClearAppearance.setCompositingMode(wallClearCompositeMode);
417    
418            // build the normal wall appearance
419            wallAppearance.setPolygonMode(wallPolygonMode);
420    
421            // load the wall texture
422            Image wallTextureImage = MazeMIDlet.makeImage("/com/nokia/midp/examples/m3g/maze3d/content/wall.png");
423            if (wallTextureImage != null)
424            {
425              Texture2D wallTexture = null;
426              wallTexture = new Texture2D(
427                      new Image2D(Image2D.RGB, wallTextureImage));
428              // the texture is repeated
429              wallTexture.setWrapping(Texture2D.WRAP_REPEAT,
430                                                              Texture2D.WRAP_CLAMP);
431              wallTexture.setBlending(Texture2D.FUNC_REPLACE);
432              wallTexture.setFiltering(Texture2D.FILTER_LINEAR,
433                                                               Texture2D.FILTER_NEAREST);
434              // set the texture
435              wallAppearance.setTexture(0, wallTexture);
436              wallClearAppearance.setTexture(0, wallTexture);
437            }
438    
439            // The maze create the planes and they are added to the world
440            Enumeration wallsEnum = maze.createPlanes();
441            while (wallsEnum.hasMoreElements())
442            {
443              Mesh wallMesh = ((Plane) wallsEnum.nextElement()).createMesh();
444              wallMesh.setAppearance(0, wallAppearance);
445              world.addChild(wallMesh);
446            }
447    
448      }

450      /**
451       * decides whether it is allowed to move to a given place
452       * otherwise it is possible to cross walls and go outside the maze
453       * @param stepx the attempted x step
454       * @param stepz the attempted z step
455       * @param forward true if direction is forward
456       * @return true if allowed to move to a given place
457       */
458      private boolean canMove(float stepx, float stepz, boolean forward)
459      {
460            float x = locationx + stepx;
461            float z = locationz + stepz;
462            // First check if inside the maze area
463            float mazeSize = MAZE_SIDE_LENGTH / 2 - 2;
464            if (x <= -mazeSize
465                    || z <= -mazeSize
466                    || x > mazeSize
467                    || z > mazeSize)
468            {
469              return false;
470            }
471            // collision with walls is easily implemented using picking
472            // if there is something in front of the camera and the distance is too
473            // small, the movement is not allowed
474            RayIntersection ri = new RayIntersection();
475            // inverse the point of view to detect if it is moving
476            // bacwards
477            if (!forward)
478            {
479              camera.postRotate(180, 0f, 1f, 0f);
480            }
481            if (world.pick(-1, 0.5f, 0.5f, camera, ri))
482            {
483              Node selected = ri.getIntersected();
484              if (selected instanceof Mesh)
485              {
486                    // multiply per 9 since that's the size of a step
487                    float distance = ri.getDistance() * 9;
488    
489                    if (distance <= 0.1)
490                    {
491                      return false;
492                    }
493              }
494            }
495            if (!forward)
496            {
497              // revert to the original view
498              camera.postRotate(-180, 0f, 1f, 0f);
499            }
500            return true;
501      }

503      /**
504       * Method checks whenever the keys are pressed
505       */
506      private void checkKeys()
507      {
508            int keyState = getKeyStates();
509            boolean moved = false;
510    
511            // if fire is detected the mesh in front of you is made semi transparent
512            if ((keyState & FIRE_PRESSED) != 0)
513            {
514              RayIntersection ri = new RayIntersection();
515              if (world.pick(-1, 0.5f, 0.5f, camera, ri))
516              {
517                    // if there is something to pick
518                    Node selected = ri.getIntersected();
519                    if (selected instanceof Mesh)
520                    {
521                      // make the wall semi transparent
522                      transparentMesh = ((Mesh) selected);
523                      transparentMesh.setAppearance(0, wallClearAppearance);
524                      transparentMesh.setAlphaFactor(0.8f);
525                    }
526              }
527            }
528            // here we assume only one button is pressed otherwise the "second"
529            // press is ignored
530            if ((keyState & LEFT_PRESSED) != 0)
531            {
532              // move to the left
533              angley += ANGLE_STEP;
534              stepx = STEP * (float) Math.sin(Math.toRadians(angley));
535              stepz = STEP * (float) Math.cos(Math.toRadians(angley));
536              // shift the background a bit to the right
537              background.setCrop(background.getCropX() + 3,
538                                                     0,
539                                                     getWidth(),
540                                                     getHeight());
541              moved = true;
542            }
543            else if ((keyState & RIGHT_PRESSED) != 0)
544            {
545              // move to the right
546              angley -= ANGLE_STEP;
547              stepx = STEP * (float) Math.sin(Math.toRadians(angley));
548              stepz = STEP * (float) Math.cos(Math.toRadians(angley));
549              // shift the background a bit to the left
550              background.setCrop(background.getCropX() - 3,
551                                                     0,
552                                                     getWidth(),
553                                                     getHeight());
554              moved = true;
555            }
556            else if ((keyState & UP_PRESSED) != 0)
557            {
558              // move forward, but first check if it is allowed
559              if (canMove(-stepx, -stepz, true))
560              {
561                    locationx -= stepx;
562                    locationz -= stepz;
563              }
564              moved = true;
565            }
566            else if ((keyState & DOWN_PRESSED) != 0)
567            {
568              // move backwards, but first check if it is allowed
569              if (canMove(stepx, stepz, false))
570              {
571                    locationx += stepx;
572                    locationz += stepz;
573              }
574              moved = true;
575            }
576    
577            // if moved we make the last transparent wall normal
578            // and reset the view
579            if (moved)
580            {
581              if (!finished && maze.isAtTheEnd(locationx, locationz, 10f))
582              {
583                    finished = true;
584              }
585              // if there was a mesh semi transparent
586              // return to normal
587              if (transparentMesh != null)
588              {
589                    transparentMesh.setAppearance(0, wallAppearance);
590                    transparentMesh.setAlphaFactor(1f);
591                    transparentMesh = null;
592              }
593              // if we move we reset the top view state
594              if (topView)
595              {
596                    switchView();
597              }
598              setupCamera();
599            }
600      }

625      /**
626       * Draw 2d information on top of the screen
627       * @param g the graphics object
628       */
629      private void draw2D(Graphics g)
630      {
631           // text color
632            g.setColor(45, 235, 45);
633            StringBuffer status = new StringBuffer("");
634            if (finished)
635            {
636              status.append("Done");
637            }
638            else
639            {
640              duration = System.currentTimeMillis() - gameStart;
641            }
642            status.append(" Time:").append(duration / 1000);
643    
644            // draw time on the top right
645            g.drawString(status.toString(),
646                                     getWidth(),
647                                     0,
648                                     Graphics.TOP | Graphics.RIGHT);
649            // draw FPS on the bottom left
650            g.drawString("FPS: " + fps,
651                                     0,
652                                     getHeight(),
653                                     Graphics.BOTTOM | Graphics.LEFT);
654      }
655    
656      /**
657       * paints the scene
658       * @param g graphics object
659       */
660      private void draw3D(Graphics g)
661      {
662            boolean bound = false;
663            try
664            {
665              // binds the target
666              g3d.bindTarget(g);
667              bound = true;
668              // advance the animation
669              world.animate((int)(System.currentTimeMillis() - gameStart));
670              // do the rendering
671              g3d.render(world);
672            }
673            finally
674            {
675              // release the target
676              if (bound)
677              {
678                    g3d.releaseTarget();
679              }
680            }
681      }

684      /**
685       * The threads run method
686       */
687      public void run()
688      {
689            Graphics g = getGraphics();
690            while (playing)
691            {
692              try
693              {
694                    long startTime = System.currentTimeMillis();
695                    if (isShown())
696                    {
697                      // update the world if the player has moved
698                      checkKeys();
699                      long drawingTime = System.currentTimeMillis();
700    
701                      // draw the 3d part
702                      draw3D(g);
703                      drawingTime = System.currentTimeMillis() - drawingTime;
704                      // calculate fps
705                      fps = (drawingTime <= 1) ? 1000 : 1000 / drawingTime;
706    
707                      // draw the 2d part
708                      draw2D(g);
709                      // flush to the screen
710                      flushGraphics();
711                    }
712                    // wait for a little and give the other threads
713                    // the chance to run
714                    long timeTaken = System.currentTimeMillis() - startTime;
715                    synchronized (this)
716                    {
717                      if (timeTaken < loopDuration)
718                      {
719                            wait(loopDuration - timeTaken);
720                            if (loopDuration > MILLIS_PER_TICK)
721                            {
722                              loopDuration--;
723                            }
724                      }
725                      else
726                      {
727                        loopDuration = timeTaken;
728                        wait(1);
729                      }
730                    }
731              }
732              catch (Exception e)
733              {
734                    // ignore. Not much to but we want to keep the loop running
735              }
736            }
737      }

740      /**
741       * It is possible that the canvas size may change and the application
742       * must be prepared to resize itself accordingly.  This routine just
743       * calls application methods stop, init to handle this
744       * change in conditions.
745       * @param w width
