002    package com.nokia.midp.examples.m3g.maze3d;
003    
004    import java.util.Enumeration;
005    import java.util.Hashtable;
006    import javax.microedition.lcdui.*;
007    import javax.microedition.m3g.*;
008    
009    /**
010     * A class that contains a list of 3D graphics properties.
011     */
012    class Graphics3DProperties
013      extends List
014      implements CommandListener
015    {
016      /** A reference to the MazeMIDlet */
017      private final MazeMIDlet midlet;
018    
019      /**
020       * Constructs the Graphics3DProperties and initializes properties and keys
021       * @param midlet the client MazeMIDlet
022       */
023      Graphics3DProperties(MazeMIDlet midlet)
024      {
025            super("Graphics 3D Properties", List.IMPLICIT);
026    
027            this.midlet = midlet;
028    
029            Hashtable props = Graphics3D.getProperties();
030            Enumeration propKeys = props.keys();
031            while (propKeys.hasMoreElements())
032            {
033              Object key = propKeys.nextElement();
034              append(key.toString() + ": " + props.get(key).toString(), null);
035            }
036            // some entries are too long
037            setFitPolicy(List.TEXT_WRAP_ON);
038    
039            addCommand(new Command("Back", Command.BACK, 1));
040            setCommandListener(this);
041      }
042    
043      /**
044       * This method invokes the MIDlet's showMenu method.
045       * @param command not used
046       * @param displayable not used
047       */
048      public void commandAction(Command command, Displayable displayable)
049      {
050            midlet.showMenu();
051      }
052    
053    }
