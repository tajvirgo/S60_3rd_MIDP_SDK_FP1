002    package com.nokia.midp.examples.JSR75.FileHandler;
003    
004    import javax.microedition.io.file.FileSystemListener;
005    import javax.microedition.io.file.FileSystemRegistry;
006    
007    
008    /**
009    FileHandlerModelFileSystemListener is a thread class that
010    is used for receiving status
011    notification when adding or removing a
012    file system root. This can be achieved
013    by inserting or removing a card from a
014    device or by mounting or unmounting file
015    systems to a device.
016    */
017    
018    class FileHandlerModelFileSystemListener extends Thread implements FileSystemListener
019    {
020        // Default classes contructor
021        public FileHandlerModelFileSystemListener( FileHandlerViewController aMidlet )
022        {
023            this.iMidlet = aMidlet;
024        }
025    
026        // Threads starting point is Run().
027        public void run()
028        {
029            // Add this class to listen to volume changes.
030            FileSystemRegistry.addFileSystemListener( this );
031        }
032    
033        //  This method is invoked when a root on the
034        // device has changed state.
035        public void rootChanged(int state, String rootName)
036        {
037            if ( state == 1 )
038            {
039                // Clear screen and print directly to.
040                iMidlet.deleteAll();
041                iMidlet.append("Root structure Changed. '" + rootName +"' has been removed.\n",null);
042            }
043            else
044            {
045                // Clear screen and print directly to.
046                iMidlet.deleteAll();
047                iMidlet.append("Root structure Changed. '" + rootName +"' has been added.\n",null);
048                }
049    
050            // Call mainäs method to update mounted volumelist.
051            iMidlet.upDateVolumes();
052        }
053        // Classe's members are here.
054        private final FileHandlerViewController iMidlet;
055    }// End of file
