002    package com.nokia.midp.examples.JSR75.FileHandler;
003    
004    import java.lang.Exception;
005    
006    /**
007    FileHandlerErrorHandler class takes care of the exceptions
008    others fall into.
009    */
010    
011    public final class FileHandlerErrorHandler
012    {
013        // Default classes contructor
014        public FileHandlerErrorHandler(FileHandlerViewController aMidlet)
015        {
016            this.iMidlet = aMidlet;
017        }
018    
019        // Static method to print the exception to user
020        public static void handleError( Exception e,
021                                        FileHandlerViewController aMidlet )
022        {
023            e.printStackTrace();
024            FileHandlerErrorHandler instance = new FileHandlerErrorHandler( aMidlet );
025            instance.PrintError( e );
026        }
027    
028        // Print is done actually here.
029        private void PrintError( Exception e )
030        {
031            // Clear controller's listBox's view.
032            iMidlet.deleteAll();
033            iMidlet.append( "An error has occured: " +e.toString(),null );
034            // Restore Conenctionpoint to USED_ROOT.
035            try {
036              iMidlet.OpenFileConn();
037            }
038            catch(Exception e2)
039            {
040              // silent here (exception condition already reported
041            }
042        }
043        // Classe's members are here.
044        private final FileHandlerViewController iMidlet;
045    }// End of file
