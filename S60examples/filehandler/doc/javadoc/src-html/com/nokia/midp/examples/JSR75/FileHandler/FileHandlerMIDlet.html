002    package com.nokia.midp.examples.JSR75.FileHandler;
003    
004    import javax.microedition.midlet.*;
005    import javax.microedition.lcdui.Display;
006    
007    /**
008    FileHandlerMIDlet is where the application on created and
009    run. Application management software contructs this and calls
010    startApp() method.
011    */
012    public class FileHandlerMIDlet extends MIDlet
013    {
014    
015       // Default classes contructor
016       public FileHandlerMIDlet()
017        {
018            instance = this;
019            displayable = new FileHandlerViewController( this );
020        }
021    
022        public void startApp()
023        {
024            // Get current's manager of the display and input
025            // devices of the system. Requests that a different
026            // Displayable object be made visible on the display.
027            // Show FileHandlerViewController class.
028            Display.getDisplay( this ).setCurrent( displayable );
029        }
030    
031        public void pauseApp()
032        {
033            // Notifies the application management software
034            // that the MIDlet does not want to be
035            // active and has entered the Paused state
036            instance.notifyPaused();
037        }
038    
039        public void destroyApp( boolean unconditional )
040        {}
041    
042        public void quitApp()
043        {
044            // Close the FileConnection
045            displayable.closeFileConn();
046    
047            // Destory the application
048            instance.destroyApp( true );
049    
050            // Used by an MIDlet to notify the application
051            // management software that it has entered into the Destroyed state
052            instance.notifyDestroyed();
053    
054            instance = null;
055        }
056    
057        // Classe's members are here.
058        private static FileHandlerMIDlet instance;
059        private FileHandlerViewController displayable = null;
060    }// End of file
