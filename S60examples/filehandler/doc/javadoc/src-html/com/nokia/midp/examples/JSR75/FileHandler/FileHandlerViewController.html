063        private void PrintFileConnectionVersion()
064        {
065            // Check that the File Connection Optional Package is there
066            String v = System.getProperty( "microedition.io.file.FileConnection.version" );
067            String systemSeparator = System.getProperty( "file.separator" );
068            if (  v != null  )
069            {
070                append( "File Connection v" + v + " ", null);
071                append( "found.", null);
072                append( "File separator is", null);
073                append( "'" + systemSeparator + "'.",null );
074                append( LIST_START,null );
075            }
076            else
077            {
078                // JSR75 was not found os system. Remove Start possibility.
079                removeCommand( startCommand );
080                append( FILE_FAIL,null );
081            }
082        }

084        //Indicates that a command event has occurred on Displayable
085        public void commandAction( Command c, Displayable d )
086        {
087            // If EXIT is called..
088            if ( c == exitCommand )
089            {
090                iMidlet.quitApp();
091            }
092            else if (  c == startCommand  )
093            {
094                // Create Thread for FileListener and start it!
095                // This is done in here because Wellcome screen had to be shown first.
096                // Adding this triggers access granting prompter.
097                FileHandlerModelFileSystemListener changer = new FileHandlerModelFileSystemListener( this );
098                changer.start();
099    
100                // List all available root.
101                upDateVolumes();
102                // Remove StartCommand. No need any more
103                removeCommand( startCommand );
104    
105                //Clear ListBox's items (clears the Screen).
106                this.deleteAll();
107    
108                // Info
109    
110                this.append( LIST_SELECT,null );
111            }
112            // This is true when <select> is pressed on ListBox
113            else if ( c == List.SELECT_COMMAND )
114            {
115                // Determinate action according to UI_State
116                // 1 = Volume select, 2 = Actions.
117                switch ( UI_STATE )
118                {
119                    case ACTION_COMMANDS_STATE: // Action mode, commands are online
120                    {
121                        // Check if selected item is
122                        // Not info and Not Folder and is ends with ".TXT" litteral.
123                        if (  ( isInfoElement( this.getString( getSelectedIndex() ) )==false ) &&
124                           (  this.getString( getSelectedIndex() ).endsWith( "/" )==false ) &&
125                           (  this.getString( getSelectedIndex() ).endsWith( ".txt" ) ) )
126                        {
127                            // Start the OpenFile Thread, Selected item from listbox was
128                            // .txt file (not info or folder)
129                            FileHandlerThread_openfile OpenFileThread = new FileHandlerThread_openfile( this,fconn,this.getString( getSelectedIndex() ) );
130                            OpenFileThread.start();
131                        }
132                        else
133                        {
134                            // Selected item must be folder. Info item is check in called
135                            // function.
136                            changeDir();
137                        }
138                        break;
139                    }
140                }
141            }
142            // Handle other than select event on keyboard.
143            else if ( c != List.SELECT_COMMAND )
144            {
145                switch ( UI_STATE )
146                {
147                    case VOLUME_SELECTION_STATE:  // Volume selection State.
148                    {
149    
150                        // Set the Root to be used.
151                        setUsedRoot( c.getLabel(),0 );
152    
153                        // Store initial root
154                        initialRoot =  getUsedRoot();
155    
156                        // Open Root connection acording to USED_ROOT member.
157                        OpenFileConn();
158    
159                        if (this.fconn != null)
160                        {
161                            // Clear the listbox
162                            this.deleteAll();
163    
164                            // Info message
165                            this.append("Selected Volume is ", null);
166                            this.append("'"+c.getLabel()+"'.",null );
167                            // List all command from model.
168                            iModel.ListActionCommands();
169    
170                            // Shift state to Action from Volume
171                            UI_STATE=ACTION_COMMANDS_STATE;
172                        }
173                        break;
174                    }
175                    case ACTION_COMMANDS_STATE: // action commands
176                    {
177    
178                        // "Back" Action was selected on Commands.
179                        if ( ( String )iModel.GetCommandVector().elementAt( 0 ) == c.getLabel() )
180                        {
181                            // Go back in roots. Params in "ListDirs after"
182                            upOneLevel( true );
183                        }
184    
185                        // "List Directory" Action was selected on Commands.
186                        if ( ( String )iModel.GetCommandVector().elementAt( 1 ) == c.getLabel() )
187                        {
188                            // Initialize thread and launch it.
189                            FileHandlerThread_List listThread = new FileHandlerThread_List( this,fconn );
190                            listThread.start();
191                        }
192    
193                        // "Open & Edit file" Action was selected on Commands.
194                        if ( ( String )iModel.GetCommandVector().elementAt( 2 ) == c.getLabel() )
195                        {
196                            // Check wheater the selected item on listbox is text file.
197                            if ( ( isInfoElement( this.getString( getSelectedIndex() ) )==false ) &&
198                                 ( this.getString( getSelectedIndex() ).endsWith( "/" )==false ) &&
199                                 ( this.getString( getSelectedIndex() ).endsWith( ".txt" ) ) )
200                            {
201                                // Initialize thread and launch it.
202                                FileHandlerThread_openfile OpenFileThread = new FileHandlerThread_openfile( this,fconn,this.getString( getSelectedIndex() ) );
203                                OpenFileThread.start();
204                            }
205                            else
206                            {
207                                // Info message to listbox
208                                append( NOT_TEXT_FILE,null );
209                            }
210                        }
211    
212                        // "Change Folder" Action was selected on Commands.
213                        if ( ( String )iModel.GetCommandVector().elementAt( 3 ) == c.getLabel() )
214                        {
215                            changeDir();
216                        }
217    
218                        // "Create Folder" Action was selected on Commands.
219                        if ( ( String )iModel.GetCommandVector().elementAt( 4 ) == c.getLabel() )
220                        {
221                            // Show the Dialog for MakeDir, zero is functionality code (makedir)
222                            // 50 param in MaxLength on the TextBox.
223                            iDialogs.shotTextBox( MKDIR_DTITLE,DIR_NAME+DIR_SEPARATOR,50,0 );
224                        }
225    
226                        // "Rename" Action was selected on Commands.
227                        if ( ( String )iModel.GetCommandVector().elementAt( 5 ) == c.getLabel() )
228                        {
229                            // Check if selected item is Info item.
230                            if ( isInfoElement( this.getString( getSelectedIndex() ) )==false )
231                            {
232                                // Show the Dialog for MakeDir, 1 is functionality code (rename)
233                                // 50 param in MaxLength on the TextBox.
234                                iDialogs.shotTextBox( REN_DTITLE,this.getString( getSelectedIndex() ),50,1 );
235                            }
236                            else
237                            {
238                                // Info message to listbox
239                                append( NOTHING_TO_DO,null );
240                            }
241                        }
242    
243                        // "Permissions" Action was selected on Commands.
244                        if ( ( String )iModel.GetCommandVector().elementAt( 6 ) == c.getLabel() )
245                        {
246                            // Check if selected item is Info item.
247                            if ( isInfoElement( this.getString( getSelectedIndex() ) )==false )
248                            {
249                                // Initialize thread and launch it.
250                                FileHandlerThread_permissions permissionsThread = new FileHandlerThread_permissions( this,fconn,this.getString( getSelectedIndex() ) );
251                                permissionsThread.start();
252                            }
253                            else
254                            {
255                                // Info message to listbox
256                                append( NOTHING_TO_DO,null );
257                            }
258                        }
259    
260                        // "Size" Action was selected on Commands.
261                        if ( ( String )iModel.GetCommandVector().elementAt( 7 ) == c.getLabel() )
262                        {
263                            // Check if selected item is Info item.
264                            if ( isInfoElement( this.getString( getSelectedIndex() ) )==false )
265                            {
266                                // Initialize thread and launch it.
267                                FileHandlerThread_fileSize fileSizeThread = new FileHandlerThread_fileSize( this,fconn,this.getString( getSelectedIndex() ) );
268                                fileSizeThread.start();
269                            }
270                            else
271                            {
272                                // Info message to listbox
273                                append( NOTHING_TO_DO,null );
274                            }
275                        }
276    
277                        // "URL" Action was selected on Commands.
278                        if ( ( String )iModel.GetCommandVector().elementAt( 8 ) == c.getLabel() )
279                        {
280                            // Check if selected item is Info item.
281                            // There is one controol boolean print/update member root variable.
282                            if ( isInfoElement( this.getString( getSelectedIndex() ) )==false )
283                            {
284                                // Initialize thread and launch it.
285                                // The last boolena param is wheater the result is printed or
286                                // is USED_ROOT setted according to listbox's selection
287                                FileHandlerThread_GetURL getURLThread = new FileHandlerThread_GetURL( this,fconn,this.getString( getSelectedIndex() ),true );
288                                getURLThread.start();
289                            }
290                            else
291                            {
292                                // Info message to listbox
293                                append( NOTHING_TO_DO,null );
294                            }
295                        }
296    
297                        // "Delete" Action was selected on Commands.
298                        if ( ( String )iModel.GetCommandVector().elementAt( 9 ) == c.getLabel() )
299                        {
300                            // Check if selected item is Info item.
301                            if ( isInfoElement( this.getString( getSelectedIndex() ) )==false )
302                            {
303                                // Initialize thread and launch it.
304                                FileHandlerThread_delete deleteThread = new FileHandlerThread_delete( this,fconn,this.getString( getSelectedIndex() ) );
305                                deleteThread.start();
306                            }
307                            else
308                            {
309                                // Info message to listbox
310                                append( NOTHING_TO_DO,null );
311                            }
312                        }
313    
314                        // "Used Mem" Action was selected on Commands.
315                        if (  ( String )iModel.GetCommandVector().elementAt( 10 ) == c.getLabel() )
316                        {
317                            // Initialize thread and launch it.
318                            FileHandlerThread_usedSize usedSizeThread = new FileHandlerThread_usedSize( this,fconn );
319                            usedSizeThread.start();
320                        }
321    
322                        // "Free size" Action was selected on Commands.
323                        if ( ( String )iModel.GetCommandVector().elementAt( 11 ) == c.getLabel() )
324                        {
325                            // Initialize thread and launch it.
326                            FileHandlerThread_AvailSize availSizeThread = new FileHandlerThread_AvailSize( this,fconn );
327                            availSizeThread.start();
328                        }
329    
330                        // "Select volume" Action was selected on Commands.
331                        if (  ( String )iModel.GetCommandVector().elementAt( 12 ) == c.getLabel() )
332                        {
333                            // Clear listbox
334                            this.deleteAll();
335                            upDateVolumes();
336                        }
337                    break;
338                    } // End of Case 2.
339                }// End of Switch
340            } // End of Else If (Handle other than select event on keyboard.)
341        } // EOClass

344        // Create menu view according to vector member. This is contructed in Model.
345        public void createMenuView( Enumeration menuItems )
346        {
347            int index;
348            // Delete All Command from Displayable surface.
349            for ( index=0;index<commandVector.size();index++ )
350            {
351                // Remove all command from displayable.
352                // Command are from Vector.
353                removeCommand( ( Command )commandVector.elementAt( index ) );
354            }
355            // Flush the local CommandVector
356            commandVector.removeAllElements();
357    
358            if ( menuItems == null )
359            {}
360            else
361            {
362                // Reset
363                index=0;
364                // Add New commands to vector
365                while (  menuItems.hasMoreElements() )
366                {
367                    // Add Commands from Model or from Volume List to displayable.
368                    commandVector.addElement( new Command( ( String )menuItems.nextElement(), Command.SCREEN, 1 ) );
369                    // Add commands also to vector.
370                    addCommand( ( Command )commandVector.elementAt( index ) );
371                    index++;
372                }
373            }
374        }

406        public void OpenFileConn()
407        {
408            try
409            {
410                // Close previous connection first
411                closeFileConn();
412                // Open FileConnection accrording to member
413                this.fconn = ( FileConnection )Connector.open( getUsedRoot() );
414            }
415            catch(java.lang.Exception se)
416            {
417                  // Get roots from FileSystemRegistry interface's method.
418    
419                  Enumeration rootEnum = FileSystemRegistry.listRoots();
420    
421                  // Create view of available menu
422                  createMenuView(rootEnum);
423    
424                  setUsedRoot(null, 0);
425    
426                  deleteAll();
427                  append(se.getMessage()+ "\n",null);
428                  append("Select Volume", null);
429    
430                  UI_STATE = VOLUME_SELECTION_STATE;
431            }
432        }

473        // Changes the current connection pointing directory
474        private void changeDir()
475        {
476            // Chck if listbox's item ends with / (folder)
477            // JSR75 API also has isDirectory() method to do this in thread.
478            if ( ( this.getString( getSelectedIndex() ).endsWith( "/" )==true ) )
479            {
480                // Set the member USED_ROOT with append param (1).
481                setUsedRoot( this.getString( getSelectedIndex() ),1 );
482    
483                // Open Root connection
484                OpenFileConn();
485    
486                if (this.fconn == null)
487                  return;
488                // List dirs
489                FileHandlerThread_List listThread = new FileHandlerThread_List( this,fconn );
490                listThread.start();
491            }
492        }

494        // Go back one level
495        private void upOneLevel( boolean aListDirectory )
496        {
497            try
498            {
499                // pass parent indication ( .. ) to GetURL method with flag that controll's the thread
500                // so that no result is printed on screen (false)
501                FileHandlerThread_GetURL getURLThread = new FileHandlerThread_GetURL( this,fconn,"..",false );
502                getURLThread.start();
503    
504                // Is list directory applied after?
505                if ( aListDirectory )
506                {
507    
508                    // If we are going to display the directory contents wait
509                    // for get thread to complete so that we display the correct
510                    // contents.
511                    while (getURLThread.isAlive()) Thread.sleep(500);
512    
513                    // If an error occured during the execution of the thtread the
514                    // UI_STATE is changed.
515                    if (UI_STATE == VOLUME_SELECTION_STATE)
516                       return;
517    
518                    // Get the refreshed list
519                    FileHandlerThread_List listThread = new FileHandlerThread_List( this,fconn );
520                    listThread.start();
521                }
522            }
523                // Catch all exceptions and forward them on.
524            catch ( Exception e )
525            {
526                FileHandlerErrorHandler.handleError( e,this );
527            }
528        }

558        // This method checks if given para ends with information mark, dot.
559        private boolean isInfoElement( String aString )
560        {
561            // the purpose of this fuction is to check wheater given string ends with dot.
562            // If so, Listbox's element at hand is charaterized as information, therefore
563            // No thread is to rise.
564            if ( aString.endsWith( "." )==true )
565            {
566                return true;
567            }
568            else
569            {
570                return false;
571            }
572        }

574        // This is called when TextBox dialog end wih DONE.
575        public void AfterDialog( String aText, int aFunction )
576        {
577            // Get Current root
578            String aCurrentRoot = getUsedRoot();
579    
580            // This is called after each dialog based functionality.
581            switch ( aFunction )
582            {
583                case 0:    // MakeDir command
584                {
585                    // Set the New folder connection
586                    setUsedRoot( aText,1 );
587                    // open connection
588                    OpenFileConn();
589                    if (this.fconn == null)
590                          break;
591    
592                    // Create new Filder according to user's input.
593                    FileHandlerThread_makeDir makeDirThread = new FileHandlerThread_makeDir( this,fconn,aText,aCurrentRoot );
594                    makeDirThread.start();
595                    break;
596                }
597                case 1:  // Rename command
598                {
599                    // Set the Connection to current file/folder ( selected item )
600                    setUsedRoot( this.getString( getSelectedIndex() ),1 );
601    
602                    // open connection
603                    OpenFileConn();
604    
605                    if (this.fconn == null)
606                          break;
607    
608                    // Call the Rename thread
609                    FileHandlerThread_rename renameThread = new FileHandlerThread_rename( this,fconn,aText,this.getString( getSelectedIndex() ),aCurrentRoot );
610                    renameThread.start();
611                    break;
612                }
613                case 2: // Call Write
614                {
615                    // Initialize the WriteFile thread and launch it.
616                    FileHandlerThread_writefile WriteFileThread = new FileHandlerThread_writefile( this,fconn,this.getString( getSelectedIndex() ),aText );
617                    WriteFileThread.start();
618                }
619            }
620        }
